33  /**
34   * Sea Glass TabbedPaneTabAreaPainter. Does nothing.
35   */
36  public final class TabbedPaneTabAreaPainter extends AbstractRegionPainter {
37  
38      /**
39       * DOCUMENT ME!
40       *
41       * @author  $author$
42       * @version $Revision$, $Date$
43       */
44      public static enum Which {
45          BACKGROUND_ENABLED_TOP, BACKGROUND_ENABLED_LEFT, BACKGROUND_ENABLED_BOTTOM, BACKGROUND_ENABLED_RIGHT,
46          BACKGROUND_DISABLED_TOP, BACKGROUND_DISABLED_LEFT, BACKGROUND_DISABLED_BOTTOM, BACKGROUND_DISABLED_RIGHT,
47      }
48  
49      private Color        baseColor      = decodeColor("toolbarToggleButtonBase");
50      private TwoColors    tabLineInner     = new TwoColors(baseColor, deriveColor(baseColor, 0f, 0f, 0f, 0x28));
51      private TwoColors    tabLineInnerEdge = new TwoColors(baseColor, deriveColor(baseColor, 0f, 0f, 0f, 0xf0));
52  
53      /** DOCUMENT ME! */
54      public Which         state;
55      private PaintContext ctx;
56      private boolean      isDisabled;
57  
58      /**
59       * Creates a new TabbedPaneTabAreaPainter object.
60       *
61       * @param state DOCUMENT ME!
62       */
63      public TabbedPaneTabAreaPainter(Which state) {
64          super();
65          this.state = state;
66          this.ctx   = new PaintContext(CacheMode.NO_CACHING);
67  
68          switch (state) {
69  
70          case BACKGROUND_DISABLED_TOP:
71          case BACKGROUND_DISABLED_LEFT:
72          case BACKGROUND_DISABLED_BOTTOM:
73          case BACKGROUND_DISABLED_RIGHT:
74              isDisabled = true;
75              break;
76  
77          case BACKGROUND_ENABLED_TOP:
78          case BACKGROUND_ENABLED_LEFT:
79          case BACKGROUND_ENABLED_BOTTOM:
80          case BACKGROUND_ENABLED_RIGHT:
81              isDisabled = false;
82              break;
83          }
84      }
85  
86      /**
87       * @see com.seaglasslookandfeel.painter.AbstractRegionPainter#doPaint(java.awt.Graphics2D,
88       *      javax.swing.JComponent, int, int, java.lang.Object[])
89       */
90      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
91          JTabbedPane tabPane     = (JTabbedPane) c;
92          int         orientation = tabPane.getTabPlacement();
93  
94          if (orientation == JTabbedPane.LEFT || orientation == JTabbedPane.RIGHT) {
95              paintVerticalLine(g, c, 0, height / 2, width, height);
96          } else {
97              paintHorizontalLine(g, c, 0, height / 2, width, height);
98          }
99      }
100 
101     /**
102      * @see com.seaglasslookandfeel.painter.AbstractRegionPainter#getPaintContext()
103      */
104     protected PaintContext getPaintContext() {
105         return ctx;
106     }

136     /**
137      * @param g
138      * @param width
139      * @param height
140      */
141     private void paintLine(Graphics2D g, int width, int height) {
142         Shape s = shapeGenerator.createRectangle(0, height/2, width, 1);
143         g.setPaint(getLinePaint(s, tabLineInnerEdge));
144         g.fill(s);
145 
146         s = shapeGenerator.createRectangle(0, height/2-1, width, 1);
147         g.setPaint(getLinePaint(s, tabLineInner));
148         g.fill(s);
149         
150         s = shapeGenerator.createRectangle(0, height/2+1, width, 1);
151         g.setPaint(getLinePaint(s, tabLineInner));
152         g.fill(s);
153     }

155     /**
156      * @param s
157      * @param colors
158      * @return
159      */
160     public Paint getLinePaint(Shape s, TwoColors colors) {
161         return createLineGradient(s, colors);
162     }
163 
164     /**
165      * @param s
166      * @param colors
167      * @return
168      */
169     private Paint createLineGradient(Shape s, TwoColors colors) {
170         Rectangle2D bounds = s.getBounds2D();
171         float x = (float) bounds.getX();
172         float y = (float) bounds.getY();
173         float w = (float) bounds.getWidth();
174         return createGradient(x, y, w + x, y, 
175             new float[] { 0f, 0.02f, 0.98f, 1f }, 
176             new Color[] {
177                 colors.top,
178                 colors.bottom,
179                 colors.bottom,
180                 colors.top 
181            });
182     }
