20  package com.seaglasslookandfeel.painter;
21  
22  import java.awt.Color;
23  import java.awt.Graphics2D;
24  
25  import javax.swing.JComponent;
26  import javax.swing.JToolBar;
27  
28  import com.seaglasslookandfeel.painter.AbstractRegionPainter.PaintContext.CacheMode;
29  
30  /**
31   * ToolBarPainter implementation.
32   * 
33   * Parts taken from Nimbus to draw drag handle.
34   * 
35   * @author Ken Orr
36   * @author Modified by Kathryn Huxtable for SeaGlass
37   */
38  public class ToolBarPainter extends AbstractRegionPainter {
39      public static enum Which {
40          BORDER_ENABLED,
41      };
42  
43      private PaintContext ctx;
44      private Which state;
45      private Color borderColor = decodeColor("seaGlassDropShadow");
46      private JToolBar toolbar;
47  
48      public ToolBarPainter(Which state) {
49          super();
50          this.ctx = new PaintContext(CacheMode.NO_CACHING);
51          this.state = state;
52      }
53  
54      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
55          this.toolbar = (JToolBar) c;
56          switch (state) {
57              case BORDER_ENABLED:
58                  paintBorder(g, width, height);
59                  break;
60          }
61      }
62      
63      private void paintBorder(Graphics2D g, int width, int height) {
64          g.setColor(borderColor);
65          g.drawLine(0, height, width, height);
66  //        g.setColor(borderColor.brighter());
67  //        g.drawLine(0, height - 0, width, height - 0);
68      }
69  
70      protected PaintContext getPaintContext() {
71          return ctx;
72      }
73  }
