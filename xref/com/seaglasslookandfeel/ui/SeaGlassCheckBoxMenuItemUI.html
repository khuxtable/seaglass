45  public class SeaGlassCheckBoxMenuItemUI extends SeaGlassMenuItemUI {
46  
47      public static ComponentUI createUI(JComponent c) {
48          return new SeaGlassCheckBoxMenuItemUI();
49      }
50  
51      protected String getPropertyPrefix() {
52          return "CheckBoxMenuItem";
53      }
54  
55      public void processMouseEvent(JMenuItem item, MouseEvent e, MenuElement path[], MenuSelectionManager manager) {
56          Point p = e.getPoint();
57          if (p.x >= 0 && p.x < item.getWidth() && p.y >= 0 && p.y < item.getHeight()) {
58              if (e.getID() == MouseEvent.MOUSE_RELEASED) {
59                  manager.clearSelectedPath();
60                  item.doClick(0);
61              } else {
62                  manager.setSelectedPath(path);
63              }
64          } else if (item.getModel().isArmed()) {
65              int c = path.length - 1;
66              MenuElement newPath[] = new MenuElement[c];
67              for (int i = 0; i < c; i++) {
68                  newPath[i] = path[i];
69              }
70              manager.setSelectedPath(newPath);
71          }
72      }
73  
74      void paintBackground(SeaGlassContext context, Graphics g, JComponent c) {
75          context.getPainter().paintCheckBoxMenuItemBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
76      }
77  
78      public void paintBorder(SynthContext context, Graphics g, int x, int y, int w, int h) {
79          ((SeaGlassContext) context).getPainter().paintCheckBoxMenuItemBorder(context, g, x, y, w, h);
80      }
