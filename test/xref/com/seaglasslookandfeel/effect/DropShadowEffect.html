22  import java.awt.image.BufferedImage;
23  import java.awt.image.Raster;
24  import java.awt.image.WritableRaster;

26  import java.util.Arrays;

28  /**
29   * DropShadowEffect - This effect currently only works with ARGB type buffered
30   * images.
31   *
32   * <p>Copied from Nimbus's DropShadowEffect by Jasper Potts. This was package
33   * local.</p>
34   */
35  public class DropShadowEffect extends ShadowEffect {
36  
37      // =================================================================================================================
38      // Effect Methods
39  
40      /**
41       * Get the type of this effect, one of UNDER,BLENDED,OVER. UNDER means the
42       * result of apply effect should be painted under the src image. BLENDED
43       * means the result of apply sffect contains a modified src image so just it
44       * should be painted. OVER means the result of apply effect should be
45       * painted over the src image.
46       *
47       * @return The effect type
48       */
49      @Override
50      public EffectType getEffectType() {
51          return EffectType.UNDER;
52      }
53  
54      /**
55       * Apply the effect to the src image generating the result . The result
56       * image may or may not contain the source image depending on what the
57       * effect type is.
58       *
59       * @param  src The source image for applying the effect to
60       * @param  dst The destination image to paint effect result into. If this is
61       *             null then a new image will be created
62       * @param  w   The width of the src image to apply effect to, this allow the
63       *             src and dst buffers to be bigger than the area the need
64       *             effect applied to it
65       * @param  h   The height of the src image to apply effect to, this allow
66       *             the src and dst buffers to be bigger than the area the need
67       *             effect applied to it
68       *
69       * @return Image with the result of the effect
70       */
71      @Override
72      public BufferedImage applyEffect(BufferedImage src, BufferedImage dst, int w, int h) {
73          if (src == null || (src.getType() != BufferedImage.TYPE_INT_ARGB && src.getType() != BufferedImage.TYPE_INT_ARGB_PRE)) {
74              throw new IllegalArgumentException("Effect only works with source images of type BufferedImage.TYPE_INT_ARGB. Type is "
75                                                 + src.getType());
76          }
77  
78          if (dst != null && dst.getType() != BufferedImage.TYPE_INT_ARGB) {
79              throw new IllegalArgumentException("Effect only works with destination images of type BufferedImage.TYPE_INT_ARGB.");
80          }
81  
82          // calculate offset
83          double trangleAngle = Math.toRadians(angle - 90);
84          int    offsetX      = (int) (Math.sin(trangleAngle) * distance + 0.5);
85          int    offsetY      = (int) (Math.cos(trangleAngle) * distance + 0.5);
86  
87          // calculate expanded size
88          int tmpOffX = offsetX + size;
89          int tmpOffY = offsetX + size;
90          int tmpW    = w + offsetX + size + size;
91          int tmpH    = h + offsetX + size;
92  
93          // create tmp buffers
94          int[]  lineBuf = getArrayCache().getTmpIntArray(w);
95          byte[] tmpBuf1 = getArrayCache().getTmpByteArray1(tmpW * tmpH);
96  
97          Arrays.fill(tmpBuf1, (byte) 0x00);
98          byte[] tmpBuf2 = getArrayCache().getTmpByteArray2(tmpW * tmpH);
99  
100         // extract src image alpha channel and inverse and offset
101         Raster srcRaster = src.getRaster();
102 
103         for (int y = 0; y < h; y++) {
104             int dy     = (y + tmpOffY);
105             int offset = dy * tmpW;
106 
107             srcRaster.getDataElements(0, y, w, 1, lineBuf);
108 
109             for (int x = 0; x < w; x++) {
110                 int dx = x + tmpOffX;
111 
112                 tmpBuf1[offset + dx] = (byte) ((lineBuf[x] & 0xFF000000) >>> 24);
113             }
114         }
115 
116         // blur
117         float[] kernel = EffectUtils.createGaussianKernel(size);
118 
119         EffectUtils.blur(tmpBuf1, tmpBuf2, tmpW, tmpH, kernel, size); // horizontal
120 
121         // pass
122         EffectUtils.blur(tmpBuf2, tmpBuf1, tmpH, tmpW, kernel, size); // vertical
123 
124         // pass
125         // rescale
126         float spread = Math.min(1 / (1 - (0.01f * this.spread)), 255);
127 
128         for (int i = 0; i < tmpBuf1.length; i++) {
129             int val = (int) (((int) tmpBuf1[i] & 0xFF) * spread);
130 
131             tmpBuf1[i] = (val > 255) ? (byte) 0xFF : (byte) val;
132         }
133 
134         // create color image with shadow color and greyscale image as alpha
135         if (dst == null)
136             dst = new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB);
137 
138         WritableRaster shadowRaster = dst.getRaster();
139         int            red          = color.getRed();
140         int            green        = color.getGreen();
141         int            blue         = color.getBlue();
142 
143         for (int y = 0; y < h; y++) {
144             int srcY         = y + tmpOffY;
145             int shadowOffset = (srcY - offsetY) * tmpW;
146 
147             for (int x = 0; x < w; x++) {
148                 int srcX = x + tmpOffX;
149 
150                 lineBuf[x] = tmpBuf1[shadowOffset + (srcX - offsetX)] << 24 | red << 16 | green << 8 | blue;
151             }
152 
153             shadowRaster.setDataElements(0, y, w, 1, lineBuf);
154         }
155 
156         return dst;
157     }
