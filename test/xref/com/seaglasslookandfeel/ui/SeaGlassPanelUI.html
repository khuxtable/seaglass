41  /**
42   * Sea Glass Panel UI delegate.
43   *
44   * <p>Based on SynthPanelUI, but sets the panel's opaque property to false.</p>
45   */
46  public class SeaGlassPanelUI extends BasicPanelUI implements PropertyChangeListener, SynthUI {
47      private SynthStyle style;
48  
49      private boolean originalOpacity;
50  
51      /**
52       * Create a UI delegate.
53       *
54       * @param  c the component for the delegate.
55       *
56       * @return the UI delegate.
57       */
58      public static ComponentUI createUI(JComponent c) {
59          return new SeaGlassPanelUI();
60      }

62      /**
63       * @see javax.swing.plaf.basic.BasicPanelUI#installUI(javax.swing.JComponent)
64       */
65      public void installUI(JComponent c) {
66          JPanel p = (JPanel) c;
67  
68          super.installUI(c);
69          installListeners(p);
70      }
71  
72      /**
73       * @see javax.swing.plaf.basic.BasicPanelUI#uninstallUI(javax.swing.JComponent)
74       */
75      public void uninstallUI(JComponent c) {
76          JPanel p = (JPanel) c;
77  
78          uninstallListeners(p);
79          super.uninstallUI(c);
80      }
81  
82      /**
83       * Install the listeners.
84       *
85       * @param p the panel.
86       */
87      protected void installListeners(JPanel p) {
88          p.addPropertyChangeListener(this);
89      }
90  
91      /**
92       * Uninstall the listeners.
93       *
94       * @param p the panel.
95       */
96      protected void uninstallListeners(JPanel p) {
97          p.removePropertyChangeListener(this);
98      }

100     /**
101      * @see javax.swing.plaf.basic.BasicPanelUI#installDefaults(javax.swing.JPanel)
102      */
103     protected void installDefaults(JPanel p) {
104         this.originalOpacity = p.isOpaque();
105         updateStyle(p);
106     }
107 
108     /**
109      * @see javax.swing.plaf.basic.BasicPanelUI#uninstallDefaults(javax.swing.JPanel)
110      */
111     protected void uninstallDefaults(JPanel p) {
112         SeaGlassContext context = getContext(p, ENABLED);
113 
114         style.uninstallDefaults(context);
115         context.dispose();
116         style = null;
117 
118         // Restore original opacity if not changed by the code.
119         LookAndFeel.installProperty(p, "opaque", originalOpacity);
120     }
121 
122     /**
123      * Update the Synth style if a property changes.
124      *
125      * @param c the panel.
126      */
127     private void updateStyle(JPanel c) {
128         SeaGlassContext context = getContext(c, ENABLED);
129 
130         style = SeaGlassLookAndFeel.updateStyle(context, this);
131         context.dispose();
132 
133         // Set the opacity according to whether the background has been set.
134         // Don't set it if the user has already set it.
135         LookAndFeel.installProperty(c, "opaque", !(c.getBackground() instanceof UIResource));
136     }
137

141     public SeaGlassContext getContext(JComponent c) {

145     /**

148      * @param  c     the panel.

152      */
153     private SeaGlassContext getContext(JComponent c, int state) {
154         return SeaGlassContext.getContext(SeaGlassContext.class, c, SeaGlassLookAndFeel.getRegion(c), style, state);
155     }
156 
157     /**
158      * Get the Synth state of the panel.
159      *
160      * @param  c the panel.

162      * @return the state.

164     private int getComponentState(JComponent c) {
165         return SeaGlassLookAndFeel.getComponentState(c);
166     }

168     /**
169      * @see javax.swing.plaf.ComponentUI#update(java.awt.Graphics, javax.swing.JComponent)
170      */
171     public void update(Graphics g, JComponent c) {
172         SeaGlassContext context = getContext(c);
173 
174         SeaGlassLookAndFeel.update(context, g);
175         context.getPainter().paintPanelBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
176         paint(context, g);
177 
178         context.dispose();
179     }
180 
181     /**
182      * @see javax.swing.plaf.ComponentUI#paint(java.awt.Graphics, javax.swing.JComponent)
183      */
184     public void paint(Graphics g, JComponent c) {
185         SeaGlassContext context = getContext(c);
186 
187         paint(context, g);
188         context.dispose();
189     }

210     /**
211      * @see java.beans.PropertyChangeListener#propertyChange(java.beans.PropertyChangeEvent)
212      */
213     public void propertyChange(PropertyChangeEvent pce) {
214         if (SeaGlassLookAndFeel.shouldUpdateStyle(pce)) {
215             updateStyle((JPanel) pce.getSource());
216         } else if (pce.getPropertyName() == "background") {
217             updateStyle((JPanel) pce.getSource());
218         }
219     }
