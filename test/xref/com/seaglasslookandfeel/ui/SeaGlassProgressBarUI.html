103     private void updateStyle(JProgressBar c) {
104         SeaGlassContext context = getContext(c, ENABLED);
105         // SynthStyle oldStyle = style;
106         style = SeaGlassLookAndFeel.updateStyle(context, this);
107         setCellLength(style.getInt(context, "ProgressBar.cellLength", 1));
108         setCellSpacing(style.getInt(context, "ProgressBar.cellSpacing", 0));
109         progressPadding = style.getInt(context, "ProgressBar.progressPadding", 0);
110         paintOutsideClip = style.getBoolean(context, "ProgressBar.paintOutsideClip", false);
111         rotateText = style.getBoolean(context, "ProgressBar.rotateText", false);
112         tileWhenIndeterminate = style.getBoolean(context, "ProgressBar.tileWhenIndeterminate", false);
113         trackThickness = style.getInt(context, "ProgressBar.trackThickness", 19);
114         tileWidth = style.getInt(context, "ProgressBar.tileWidth", 15);
115         bgFillColor = (Color) style.get(context, "ProgressBar.backgroundFillColor");
116         if (bgFillColor == null) {
117             bgFillColor = Color.white;
118         }
119         // handle scaling for sizeVarients for special case components. The
120         // key "JComponent.sizeVariant" scales for large/small/mini
121         // components are based on Apples LAF
122         String scaleKey = (String) progressBar.getClientProperty("JComponent.sizeVariant");
123         if (scaleKey != null) {
124             if ("large".equals(scaleKey)) {
125                 trackThickness = 24;
126                 tileWidth *= 1.15;
127             } else if ("small".equals(scaleKey)) {
128                 trackThickness = 17;
129                 tileWidth *= 0.857;
130             } else if ("mini".equals(scaleKey)) {
131                 trackThickness = 15;
132                 tileWidth *= 0.784;
133             }
134         }
135         context.dispose();
136     }

194     private Rectangle calcBounds(JProgressBar pBar) {
195         boundsRect.x = 0;
196         boundsRect.y = 0;
197         boundsRect.width = pBar.getWidth();
198         boundsRect.height = pBar.getHeight();
199         if (pBar.getOrientation() == JProgressBar.HORIZONTAL) {
200             boundsRect.height = Math.min(boundsRect.height, trackThickness);
201             boundsRect.y += (boundsRect.height - trackThickness) / 2;
202         } else {
203             boundsRect.width = Math.min(boundsRect.width, trackThickness);
204             boundsRect.x += (boundsRect.width - trackThickness) / 2;
205         }
206         return boundsRect;
207     }

229     protected void paint(SeaGlassContext context, Graphics g) {
230         JProgressBar pBar = (JProgressBar) context.getComponent();
231         Insets pBarInsets = pBar.getInsets();
232         Rectangle bounds = calcBounds(pBar);
233         // Save away the track bounds.
234         savedRect.setBounds(bounds);
235         // Subtract out any insets for the progress indicator.
236         bounds.x += pBarInsets.left + progressPadding;
237         bounds.y += pBarInsets.top + progressPadding;
238         bounds.width -= pBarInsets.left + pBarInsets.right + progressPadding + progressPadding;
239         bounds.height -= pBarInsets.top + pBarInsets.bottom + progressPadding + progressPadding;
240 
241         int size = 0;
242         boolean isFinished = false;
243         if (!pBar.isIndeterminate()) {
244             double percentComplete = pBar.getPercentComplete();
245             if (percentComplete == 1.0) {
246                 isFinished = true;
247             } else if (percentComplete > 0.0) {
248                 if (pBar.getOrientation() == JProgressBar.HORIZONTAL) {
249                     size = (int) (percentComplete * bounds.width);
250                 } else { // JProgressBar.VERTICAL
251                     size = (int) (percentComplete * bounds.height);
252                 }
253             }
254         }
255 
256         // Create a translucent intermediate image in which we can perform soft
257         // clipping.
258         GraphicsConfiguration gc = ((Graphics2D) g).getDeviceConfiguration();
259         BufferedImage img = gc.createCompatibleImage(bounds.width, bounds.height, Transparency.TRANSLUCENT);
260         Graphics2D g2d = img.createGraphics();
261 
262         // Clear the image so all pixels have zero alpha
263         g2d.setComposite(AlphaComposite.Clear);
264         g2d.fillRect(0, 0, bounds.width, bounds.height);
265 
266         // Render our clip shape into the image. Enable antialiasing to achieve
267         // a soft clipping effect.
268         g2d.setComposite(AlphaComposite.Src);
269         g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
270         g2d.setColor(bgFillColor);
271         CornerSize cornerSize = pBar.getOrientation() == JProgressBar.HORIZONTAL ? CornerSize.ROUND_HEIGHT : CornerSize.ROUND_WIDTH;
272         g2d.fill(shapeGenerator.createRoundRectangle(0, 0, bounds.width, bounds.height, cornerSize));
273 
274         // Use SrcAtop, which effectively uses the alpha value as a coverage
275         // value for each pixel stored in the destination. At the edges, the
276         // antialiasing of the rounded rectangle gives us the desired soft
277         // clipping effect.
278         g2d.setComposite(AlphaComposite.SrcAtop);
279 
280         // We need to redraw the background, otherwise the interior is
281         // completely white.
282         context.getPainter().paintProgressBarBackground(context, g2d, savedRect.x - bounds.x, savedRect.y - bounds.y, savedRect.width,
283             savedRect.height, pBar.getOrientation());
284         paintProgressIndicator(context, g2d, bounds.width, bounds.height, size, isFinished);
285 
286         // Dispose of the image graphics and copy our intermediate image to the
287         // main graphics.
288         g2d.dispose();
289         g.drawImage(img, bounds.x, bounds.y, null);
290 
291         if (pBar.isStringPainted()) {
292             paintText(context, g, pBar.getString());
293         }
294     }

296     /**
297      * Paint the actual internal progress bar.
298      * 
299      * @param context
300      * @param g2d
301      * @param width
302      * @param height
303      * @param size
304      * @param isFinished
305      */
306     private void paintProgressIndicator(SeaGlassContext context, Graphics2D g2d, int width, int height, int size, boolean isFinished) {
307         JProgressBar pBar = (JProgressBar) context.getComponent();
308 
309         if (tileWhenIndeterminate && pBar.isIndeterminate()) {
310             double offsetFraction = (double) getAnimationIndex() / (double) getFrameCount();
311             int offset = (int) (offsetFraction * tileWidth);
312             if (pBar.getOrientation() == JProgressBar.HORIZONTAL) {
313                 // If we're right-to-left, flip the direction of animation.
314                 if (!SeaGlassLookAndFeel.isLeftToRight(pBar)) {
315                     offset = tileWidth - offset;
316                 }
317                 // paint each tile horizontally
318                 for (int i = -tileWidth + offset; i <= width; i += tileWidth) {
319                     context.getPainter().paintProgressBarForeground(context, g2d, i, 0, tileWidth, height, pBar.getOrientation());
320                 }
321             } else {
322                 // paint each tile vertically
323                 for (int i = -offset; i < height + tileWidth; i += tileWidth) {
324                     context.getPainter().paintProgressBarForeground(context, g2d, 0, i, width, tileWidth, pBar.getOrientation());
325                 }
326             }
327         } else {
328             if (pBar.getOrientation() == JProgressBar.HORIZONTAL) {
329                 int start = 0;
330                 if (isFinished) {
331                     size = width;
332                 } else if (!SeaGlassLookAndFeel.isLeftToRight(pBar)) {
333                     start = width - size;
334                 }
335                 context.getPainter().paintProgressBarForeground(context, g2d, start, 0, size, height, pBar.getOrientation());
336             } else {
337                 // When the progress bar is vertical we always paint from bottom
338                 // to top, not matter what the component orientation is.
339                 int start = height;
340                 if (isFinished) {
341                     size = height;
342                 }
343                 context.getPainter().paintProgressBarForeground(context, g2d, 0, start, width, size, pBar.getOrientation());
344             }
345         }
346     }

348     protected void paintText(SeaGlassContext context, Graphics g, String title) {
349         if (progressBar.isStringPainted()) {
350             SynthStyle style = context.getStyle();
351             Font font = style.getFont(context);
352             FontMetrics fm = SwingUtilities2.getFontMetrics(progressBar, g, font);
353             int strLength = style.getGraphicsUtils(context).computeStringWidth(context, font, fm, title);
354             Rectangle bounds = progressBar.getBounds();
355 
356             if (rotateText && progressBar.getOrientation() == JProgressBar.VERTICAL) {
357                 Graphics2D g2 = (Graphics2D) g;
358                 // Calculate the position for the text.
359                 Point textPos;
360                 AffineTransform rotation;
361                 if (progressBar.getComponentOrientation().isLeftToRight()) {
362                     rotation = AffineTransform.getRotateInstance(-Math.PI / 2);
363                     textPos = new Point((bounds.width + fm.getAscent() - fm.getDescent()) / 2, (bounds.height + strLength) / 2);
364                 } else {
365                     rotation = AffineTransform.getRotateInstance(Math.PI / 2);
366                     textPos = new Point((bounds.width - fm.getAscent() + fm.getDescent()) / 2, (bounds.height - strLength) / 2);
367                 }
368 
369                 // Progress bar isn't wide enough for the font. Don't paint it.
370                 if (textPos.x < 0) {
371                     return;
372                 }
373 
374                 // Paint the text.
375                 font = font.deriveFont(rotation);
376                 g2.setFont(font);
377                 g2.setColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
378                 style.getGraphicsUtils(context).paintText(context, g, title, textPos.x, textPos.y, -1);
379             } else {
380                 // Calculate the bounds for the text.
381                 Rectangle textRect = new Rectangle((bounds.width / 2) - (strLength / 2),
382                     (bounds.height - (fm.getAscent() + fm.getDescent())) / 2, 0, 0);
383 
384                 // Progress bar isn't tall enough for the font. Don't paint it.
385                 if (textRect.y < 0) {
386                     return;
387                 }
388 
389                 // Paint the text.
390                 g.setColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
391                 g.setFont(font);
392                 style.getGraphicsUtils(context).paintText(context, g, title, textRect.x, textRect.y, -1);
393             }
394         }
395     }

407     @Override
408     public Dimension getPreferredSize(JComponent c) {
409         Dimension size = null;
410         Insets border = progressBar.getInsets();
411         FontMetrics fontSizer = progressBar.getFontMetrics(progressBar.getFont());
412         String progString = progressBar.getString();
413         int stringHeight = fontSizer.getHeight() + fontSizer.getDescent();
414 
415         if (progressBar.getOrientation() == JProgressBar.HORIZONTAL) {
416             size = new Dimension(getPreferredInnerHorizontal());
417             if (progressBar.isStringPainted()) {
418                 // adjust the height if necessary to make room for the string
419                 if (stringHeight > size.height) {
420                     size.height = stringHeight;
421                 }
422 
423                 // adjust the width if necessary to make room for the string
424                 int stringWidth = SwingUtilities2.stringWidth(progressBar, fontSizer, progString);
425                 if (stringWidth > size.width) {
426                     size.width = stringWidth;
427                 }
428             }
429         } else {
430             size = new Dimension(getPreferredInnerVertical());
431             if (progressBar.isStringPainted()) {
432                 // make sure the width is big enough for the string
433                 if (stringHeight > size.width) {
434                     size.width = stringHeight;
435                 }
436 
437                 // make sure the height is big enough for the string
438                 int stringWidth = SwingUtilities2.stringWidth(progressBar, fontSizer, progString);
439                 if (stringWidth > size.height) {
440                     size.height = stringWidth;
441                 }
442             }
443         }
444 
445         // handle scaling for sizeVarients for special case components. The
446         // key "JComponent.sizeVariant" scales for large/small/mini
447         // components are based on Apples LAF
448         String scaleKey = (String) progressBar.getClientProperty("JComponent.sizeVariant");
449         if (scaleKey != null) {
450             if ("large".equals(scaleKey)) {
451                 size.width *= 1.15f;
452                 size.height *= 1.15f;
453             } else if ("small".equals(scaleKey)) {
454                 size.width *= 0.90f;
455                 size.height *= 0.90f;
456             } else if ("mini".equals(scaleKey)) {
457                 size.width *= 0.784f;
458                 size.height *= 0.784f;
459             }
460         }
461 
462         size.width += border.left + border.right;
463         size.height += border.top + border.bottom;
464 
465         return size;
466     }
