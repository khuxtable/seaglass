75      public void installDefaults() {
76          if (popupMenu.getLayout() == null || popupMenu.getLayout() instanceof UIResource) {
77              popupMenu.setLayout(new DefaultMenuLayout(popupMenu, BoxLayout.Y_AXIS));
78              popupMenu.setOpaque(false);
79          }
80          updateStyle(popupMenu);
81      }
82  
83      private void updateStyle(JComponent c) {
84          SeaGlassContext context = getContext(c, ENABLED);
85          Window window = SwingUtilities.getWindowAncestor(popupMenu);
86          if (PlatformUtils.isMac() && window != null) {
87              WindowUtils.makeWindowNonOpaque(window);
88          }
89          SeaGlassStyle oldStyle = style;
90          style = (SeaGlassStyle) SeaGlassLookAndFeel.updateStyle(context, this);
91          if (style != oldStyle) {
92              if (oldStyle != null) {
93                  uninstallKeyboardActions();
94                  installKeyboardActions();
95              }
96          }
97          context.dispose();
98      }

100     protected void installListeners() {

105     protected void uninstallDefaults() {
106         SeaGlassContext context = getContext(popupMenu, ENABLED);

108         style.uninstallDefaults(context);
109         context.dispose();

111 
112         if (popupMenu.getLayout() instanceof UIResource) {
113             popupMenu.setLayout(null);
114         }
115     }
