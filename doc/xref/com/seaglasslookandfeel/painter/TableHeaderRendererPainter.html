20  package com.seaglasslookandfeel.painter;
21  
22  import java.awt.Color;
23  import java.awt.Graphics2D;
24  import java.awt.Paint;
25  import java.awt.Shape;
26  
27  import javax.swing.JComponent;
28  
29  import com.seaglasslookandfeel.painter.AbstractRegionPainter.PaintContext.CacheMode;
30  
31  /**
32   * Paint table headers.
33   */
34  public final class TableHeaderRendererPainter extends AbstractCommonColorsPainter {
35      public static enum Which {
36          BACKGROUND_DISABLED,
37          BACKGROUND_ENABLED,
38          BACKGROUND_ENABLED_FOCUSED,
39          BACKGROUND_PRESSED,
40          BACKGROUND_ENABLED_SORTED,
41          BACKGROUND_ENABLED_FOCUSED_SORTED,
42          BACKGROUND_DISABLED_SORTED,
43      }
44  
45      private Color             tableHeaderBorderEnabled       = decodeColor("tableHeaderBorderEnabled");
46      private Color             tableHeaderInteriorBaseEnabled = decodeColor("tableHeaderInteriorBaseEnabled");
47  
48      private Color             tableHeaderBorderDisabled      = disable(tableHeaderBorderEnabled);
49  
50      private Color             tableHeaderTopEnabled          = deriveColor(tableHeaderInteriorBaseEnabled, -0.027778f, -0.020842f,
51                                                                   0.035294f, 0);
52      private Color             tableHeaderUpperMidEnabled     = deriveColor(tableHeaderInteriorBaseEnabled, -0.020833f, 0.000405f,
53                                                                   -0.011765f, 0);
54      private Color             tableHeaderLowerMidEnabled     = deriveColor(tableHeaderInteriorBaseEnabled, 0f, -0.000264f, 0.007843f, 0);
55      private Color             tableHeaderBottomEnabled       = deriveColor(tableHeaderInteriorBaseEnabled, -0.020833f, -0.001033f,
56                                                                   0.031373f, 0);
57  
58      private FourColors        tableHeaderEnabled             = new FourColors(tableHeaderTopEnabled, tableHeaderUpperMidEnabled,
59                                                                   tableHeaderLowerMidEnabled, tableHeaderBottomEnabled);
60      private FourColors        tableHeaderSorted              = getCommonInteriorColors(CommonControlState.SELECTED);
61      private FourColors        tableHeaderPressed             = getCommonInteriorColors(CommonControlState.PRESSED);
62      private FourColors        tableHeaderDisabled            = disable(tableHeaderEnabled);
63      private FourColors        tableHeaderDisabledSorted      = disable(tableHeaderSorted);
64  
65      private PaintContext      ctx;
66      private CommonControlState type;
67      private boolean           isSorted;
68  
69      public TableHeaderRendererPainter(Which state) {
70          super();
71          this.ctx = new PaintContext(CacheMode.FIXED_SIZES);
72          this.type = getButtonType(state);
73          this.isSorted = (state == Which.BACKGROUND_DISABLED_SORTED || state == Which.BACKGROUND_ENABLED_SORTED || state == Which.BACKGROUND_ENABLED_FOCUSED_SORTED);
74      }
75  
76      @Override
77      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
78          Shape s = shapeGenerator.createRectangle(0, 0, width - 1, height - 1);
79          g.setPaint(getTableHeaderInteriorPaint(s, type, isSorted));
80          g.fill(s);
81  
82          g.setPaint(getTableHeaderBorderPaint(type));
83          g.drawLine(0, height - 1, width, height - 1);
84          g.drawLine(width - 1, 0, width - 1, height - 1);
85      }
86  
87      @Override
88      protected PaintContext getPaintContext() {
89          return ctx;
90      }
91  
92      private CommonControlState getButtonType(Which state) {
93          switch (state) {
94          case BACKGROUND_DISABLED:
95              return CommonControlState.DISABLED;
96          case BACKGROUND_ENABLED:
97              return CommonControlState.ENABLED;
98          case BACKGROUND_ENABLED_FOCUSED:
99              return CommonControlState.ENABLED;
100         case BACKGROUND_PRESSED:
101             return CommonControlState.PRESSED;
102         case BACKGROUND_ENABLED_SORTED:
103             return CommonControlState.ENABLED;
104         case BACKGROUND_ENABLED_FOCUSED_SORTED:
105             return CommonControlState.ENABLED;
106         case BACKGROUND_DISABLED_SORTED:
107             return CommonControlState.DISABLED;
108         }
109         return null;
110     }
111 
112     public Paint getTableHeaderBorderPaint(CommonControlState type) {
113         return type == CommonControlState.DISABLED ? tableHeaderBorderDisabled : tableHeaderBorderEnabled;
114     }
115 
116     public Paint getTableHeaderInteriorPaint(Shape s, CommonControlState type, boolean isSorted) {
117         FourColors colors = getTableHeaderColors(type, isSorted);
118         return createVerticalGradient(s, colors);
119     }
120 
121     private FourColors getTableHeaderColors(CommonControlState type, boolean isSorted) {
122         switch (type) {
123         case DISABLED:
124             return isSorted ? tableHeaderDisabledSorted : tableHeaderDisabled;
125         case ENABLED:
126             return isSorted ? tableHeaderSorted : tableHeaderEnabled;
127         case PRESSED:
128             return tableHeaderPressed;
129         }
130         return null;
131     }
