35  /**
36   * Paint a (possibly) segmented button. The default colors are suitable for
37   * drawing on a default background, for instance, a dialog box.

57       * @param state the button state.

66      /**
67       * {@inheritDoc}
68       */
69      public void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
70          SegmentType segmentStatus = getSegmentType(c);
71          int         newHeight     = getButtonHeight(c, height);
72  
73          int yOffset = (height - newHeight) / 2;
74  
75          height = newHeight;
76  
77          g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
78  
79          int x = focusInsets.left;
80          int y = focusInsets.top + yOffset;
81  
82          width  -= focusInsets.left + focusInsets.right;
83          height -= focusInsets.top + focusInsets.bottom;
84  
85          boolean useToolBarFocus = isInToolBar(c);
86          Shape   s;
87  
88          if (focused) {
89              s = createOuterFocus(segmentStatus, x, y, width, height);
90              g.setPaint(getFocusPaint(s, FocusType.OUTER_FOCUS, useToolBarFocus));
91              g.draw(s);
92              s = createInnerFocus(segmentStatus, x, y, width, height);
93              g.setPaint(getFocusPaint(s, FocusType.INNER_FOCUS, useToolBarFocus));
94              g.draw(s);
95          }
96  
97          if (!isInToolBar(c) || this instanceof TexturedButtonPainter) {
98              s = createBorder(segmentStatus, x, y, width, height);
99  
100             if (!focused) {
101                 dropShadow.fill(g, s);
102             }
103 
104             g.setPaint(getCommonBorderPaint(s, type));
105             g.fill(s);
106 
107             s = createInterior(segmentStatus, x, y, width, height);
108             g.setPaint(getCommonInteriorPaint(s, type));
109             g.fill(s);
110         }
111     }

113     /**
114      * Compute the new button height for drawing, forcing the button into a
115      * standard height.
116      *
117      * @param  c      the button.
118      * @param  height the height requested.
119      *
120      * @return the new height.
121      */
122     private int getButtonHeight(JComponent c, int height) {
123         String scaleKey  = (String) c.getClientProperty("JComponent.sizeVariant");
124         int    newHeight = 27;
125 
126         // If the key is natural, allow the button to scale.
127         if (SeaGlassStyle.NATURAL_KEY.equals(scaleKey)) {
128             return height;
129         } else if (SeaGlassStyle.LARGE_KEY.equals(scaleKey)) {
130             newHeight *= SeaGlassStyle.LARGE_SCALE; // 31
131         } else if (SeaGlassStyle.SMALL_KEY.equals(scaleKey)) {
132             newHeight *= SeaGlassStyle.SMALL_SCALE; // 23
133         } else if (SeaGlassStyle.MINI_KEY.equals(scaleKey)) {
134             newHeight *= SeaGlassStyle.MINI_SCALE; // 19
135         } else {
136             if (height < 22) {
137                 newHeight *= SeaGlassStyle.MINI_SCALE;
138             } else if (height < 26) {
139                 newHeight *= SeaGlassStyle.SMALL_SCALE;
140             } else if (height >= 30) {
141                 newHeight *= SeaGlassStyle.LARGE_SCALE;
142             }
143         }
144 
145         return newHeight;
146     }

148     /**
149      * Get the button colors for the state.
150      *
151      * @param  state the button state to paint.
152      *
153      * @return the button color set.
154      */
155     protected CommonControlState getButtonType(Which state) {
156         switch (state) {
157 
158         case BACKGROUND_DEFAULT:
159         case BACKGROUND_DEFAULT_FOCUSED:
160         case BACKGROUND_SELECTED:
161         case BACKGROUND_SELECTED_FOCUSED:
162             return CommonControlState.DEFAULT;
163 
164         case BACKGROUND_PRESSED_DEFAULT:
165         case BACKGROUND_PRESSED_DEFAULT_FOCUSED:
166             return CommonControlState.DEFAULT_PRESSED;
167 
168         case BACKGROUND_DISABLED:
169             return CommonControlState.DISABLED;
170 
171         case BACKGROUND_ENABLED:
172         case BACKGROUND_FOCUSED:
173             return CommonControlState.ENABLED;
174 
175         case BACKGROUND_PRESSED:
176         case BACKGROUND_PRESSED_FOCUSED:
177         case BACKGROUND_PRESSED_SELECTED:
178         case BACKGROUND_PRESSED_SELECTED_FOCUSED:
179             return CommonControlState.PRESSED;
180 
181         case BACKGROUND_DISABLED_SELECTED:
182             return CommonControlState.DISABLED_SELECTED;
183         }
184 
185         return null;
186     }

188     /**
189      * Get the segment type (if any) from the component's client properties.
190      *
191      * @param  c the component.
192      *
193      * @return the segment status.
194      */
195     protected SegmentType getSegmentType(JComponent c) {
196         Object      buttonType  = c.getClientProperty("JButton.buttonType");
197         SegmentType segmentType = SegmentType.NONE;
198 
199         if (buttonType != null && buttonType instanceof String && ((String) buttonType).startsWith("segmented")) {
200             String position = (String) c.getClientProperty("JButton.segmentPosition");
201 
202             if ("first".equals(position)) {
203                 segmentType = SegmentType.FIRST;
204             } else if ("middle".equals(position)) {
205                 segmentType = SegmentType.MIDDLE;
206             } else if ("last".equals(position)) {
207                 segmentType = SegmentType.LAST;
208             }
209         }
210 
211         return segmentType;
212     }

214     /**
215      * Create the shape for the outer focus ring. Designed to be drawn rather
216      * than filled.
217      *
218      * @param  segmentType the segment type.
219      * @param  x           the x offset.
220      * @param  y           the y offset.
221      * @param  w           the width.
222      * @param  h           the height.
223      *
224      * @return the shape of the outer focus ring. Designed to be drawn rather
225      *         than filled.
226      */
227     protected Shape createOuterFocus(final SegmentType segmentType, final int x, final int y, final int w, final int h) {
228         switch (segmentType) {
229 
230         case FIRST:
231             return shapeGenerator.createRoundRectangle(x - 2, y - 2, w + 3, h + 3, CornerSize.OUTER_FOCUS, CornerStyle.ROUNDED,
232                                                        CornerStyle.ROUNDED, CornerStyle.SQUARE, CornerStyle.SQUARE);
233 
234         case MIDDLE:
235             return shapeGenerator.createRectangle(x - 2, y - 2, w + 3, h + 3);
236 
237         case LAST:
238             return shapeGenerator.createRoundRectangle(x - 2, y - 2, w + 3, h + 3, CornerSize.OUTER_FOCUS, CornerStyle.SQUARE,
239                                                        CornerStyle.SQUARE, CornerStyle.ROUNDED, CornerStyle.ROUNDED);
240 
241         default:
242             return shapeGenerator.createRoundRectangle(x - 2, y - 2, w + 3, h + 3, CornerSize.OUTER_FOCUS);
243         }
244     }
