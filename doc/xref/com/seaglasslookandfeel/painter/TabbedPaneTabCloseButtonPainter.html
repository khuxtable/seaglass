46      private Color borderFocused   = decodeColor("seaGlassTabbedPaneTabCloseBorderBase", 0f, 0f, 0f, -100);

63      /**
64       * Creates a new TabbedPaneTabCloseButtonPainter object.
65       *
66       * @param state the control state to paint.
67       */
68      public TabbedPaneTabCloseButtonPainter(Which state) {
69          super();
70          this.state = state;
71          this.ctx   = new PaintContext(AbstractRegionPainter.PaintContext.CacheMode.FIXED_SIZES);
72      }

74      /**
75       * {@inheritDoc}
76       */
77      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
78          g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
79  
80          switch (state) {
81  
82          case ENABLED:
83              drawEnabledGraphic(g, width, height);
84              break;
85  
86          case MOUSEOVER:
87              drawBorder(g, width, height, borderMouseOver, 0.4f);
88              drawOverlayGraphic(g, width, height);
89              break;
90  
91          case FOCUSED:
92              drawBorder(g, width, height, borderFocused, 0.4f);
93              drawOverlayGraphic(g, width, height);
94              break;
95  
96          case PRESSED:
97              drawBorder(g, width, height, borderPressed, 0.5f);
98              drawOverlayGraphic(g, width, height);
99              break;
100         }
101     }

110     /**
111      * Draw a border around the graphic.
112      *
113      * @param g      the Graphic context.
114      * @param width  the width of the border.
115      * @param height the height of the border.
116      * @param color  the color of the border.
117      * @param size   the spread of the border from outside in, expressed as a
118      *               percentage of the height.
119      */
120     private void drawBorder(Graphics2D g, int width, int height, Color color, float size) {
121         int max        = (int) (Math.min((height - 2) * size, height / 2.0f) + 0.5);
122         int alphaDelta = color.getAlpha() / max;
123 
124         for (int i = 0; i < max; i++) {
125             Shape s        = shapeGenerator.createRoundRectangle(i, i, width - 2 * i - 1, height - 2 * i - 1, CornerSize.CHECKBOX_INTERIOR);
126             Color newColor = new Color(color.getRed(), color.getGreen(), color.getBlue(), color.getAlpha() - i * alphaDelta);
127 
128             g.setPaint(newColor);
129             g.draw(s);
130         }
131     }

133     /**
134      * Draw the "close" graphic for the simple enabled state.
135      *
136      * @param g      the Graphic context.
137      * @param width  the width of the graphic.
138      * @param height the height of the graphic.
139      */
140     private void drawEnabledGraphic(Graphics2D g, int width, int height) {
141         Shape s = shapeGenerator.createTabCloseIcon(2, 2, width - 4, height - 4);
142 
143         g.setPaint(createGraphicInnerShadowGradient(s));
144         g.fill(s);
145     }
146 
147     /**
148      * Draw the "close" graphic for a state where it overlays a border.
149      *
150      * @param g      the Graphic context.
151      * @param width  the width of the graphic.
152      * @param height the height of the graphic.
153      */
154     private void drawOverlayGraphic(Graphics2D g, int width, int height) {
155         Shape s = shapeGenerator.createTabCloseIcon(2, 2, width - 4, height - 4);
156 
157         g.setPaint(graphicBase);
158         g.fill(s);
159 
160         s = shapeGenerator.createTabCloseIcon(2, 3, width - 4, height - 4);
161         g.setPaint(createGraphicDropShadowGradient(s));
162         Shape oldClip = g.getClip();
163 
164         g.setClip(2, 3, width - 4, height - 4);
165         g.fill(s);
166         g.setClip(oldClip);
167     }

169     /**
170      * Create the gradient for the "x" inner shadow.
171      *
172      * @param  s the Shape for the gradient.
173      *
174      * @return the gradient paint.
175      */
176     private Paint createGraphicInnerShadowGradient(Shape s) {
177         Rectangle2D b    = s.getBounds2D();
178         float       midX = (float) b.getCenterX();
179         float       y1   = (float) b.getMinY();
180         float       y2   = (float) b.getMaxY();
181 
182         return createGradient(midX, y1, midX, y2, new float[] { 0f, 0.43f, 0.57f, 1f },
183                               new Color[] { graphicInnerShadow1, graphicInnerShadow2, graphicInnerShadow3, graphicInnerShadow4 });
184     }
