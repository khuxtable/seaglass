51      /**
52       * The rounding amount for a corner.
53       */
54      public enum CornerSize {
55  
56          /**
57           * Round using half the height, producing a nice quarter of a circle for
58           * the entire quarter of the rectangle. Use for horizontal lozenges that
59           * are filled.
60           */
61          ROUND_HEIGHT(0),
62  
63          /**
64           * Round using half the height plus 1, producing a nice quarter of a
65           * circle for the entire quarter of the rectangle. Use for horizontal
66           * lozenges that are drawn.
67           */
68          ROUND_HEIGHT_DRAW(0),
69  
70          /**
71           * Round using half the width, producing a nice quarter of a circle for
72           * the entire quarter of the rectangle. Use for vertical lozenges that
73           * are filled.
74           */
75          ROUND_WIDTH(0),
76  
77          /**
78           * Round using half the width plus 1, producing a nice quarter of a
79           * circle for the entire quarter of the rectangle. Use for vertical
80           * lozenges that are drawn.
81           */
82          ROUND_WIDTH_DRAW(0),
83  
84          /** Round for a generic object's interior. */
85          INTERIOR(baseRadius - 1),
86  
87          /** Round for a generic object's border. */
88          BORDER(baseRadius),
89  
90          /** Round for a generic object's inner focus ring. */
91          INNER_FOCUS(baseRadius + 1),
92  
93          /** Round for a generic object's outer focus ring. */
94          OUTER_FOCUS(baseRadius + 2),
95  
96          /** Round for a slider thumb's interior. */
97          SLIDER_INTERIOR(baseRadius - 2),
98  
99          /** Round for a slider thumb's border. */
100         SLIDER_BORDER(baseRadius - 1),
101 
102         /** Round for a slider thumb's inner focus ring. */
103         SLIDER_INNER_FOCUS(baseRadius),
104 
105         /** Round for a slider thumb's outer focus ring. */
106         SLIDER_OUTER_FOCUS(baseRadius + 1),
107 
108         /** Round for a check box's interior. */
109         CHECKBOX_INTERIOR(baseRadius / 2),
110 
111         /** Round for a check box's border. */
112         CHECKBOX_BORDER((baseRadius + 1) / 2),
113 
114         /** Round for a check box's inner focus ring. */
115         CHECKBOX_INNER_FOCUS((baseRadius + 2) / 2),
116 
117         /** Round for a check box's outer focus ring. */
118         CHECKBOX_OUTER_FOCUS((baseRadius + 3) / 2),
119 
120         /** Round for a popup menu's border. */
121         POPUP_BORDER(3),
122 
123         /** Round for a popup menu's interior. */
124         POPUP_INTERIOR(2.5),
125 
126         /** Round for a frame's border. */
127         FRAME_BORDER(baseRadius + 1),
128 
129         /** Round for a frame's inner highlight. */
130         FRAME_INNER_HIGHLIGHT(baseRadius),
131 
132         /** Round for a frame's interior. */
133         FRAME_INTERIOR(baseRadius - 1);

135         /** The rounding radius. */
136         private double radius;
137 
138         /**
139          * Create the corner size.
140          *
141          * @param radius the radius for rounding.
142          */
143         CornerSize(double radius) {
144             this.radius = radius;
145         }
146 
147         /**
148          * Return the rounding radius. Note that the {@code ROUND*} values are
149          * handled specially, as the rounding is dependent on the size of the
150          * rectangle.
151          *
152          * @param  w the width of the rectangle.
153          * @param  h the height of the rectangle.
154          *
155          * @return the radius (arc size) to use for rounding.
156          */
157         public double getRadius(int w, int h) {
158             switch (this) {
159 
160             case ROUND_HEIGHT:
161                 return h / 2.0;
162 
163             case ROUND_HEIGHT_DRAW:
164                 return (h + 1) / 2.0;
165 
166             case ROUND_WIDTH:
167                 return w / 2.0;
168 
169             case ROUND_WIDTH_DRAW:
170                 return (w + 1) / 2.0;
171 
172             default:
173                 return radius;
174             }
175         }
176     }

398     /**
399      * Return a path for a rounded internal drop shadow. This is used for
400      * progress bar tracks and search fields.
401      *
402      * @param  x the X coordinate of the upper-left corner of the shadow
403      * @param  y the Y coordinate of the upper-left corner of the shadow
404      * @param  w the width of the shadow
405      * @param  h the height of the rectangle
406      *
407      * @return a path representing the shadow.
408      */
409     public Shape createInternalDropShadowRounded(final int x, final int y, final int w, final int h) {
410         final double radius = h / 2;
411         final int    right  = x + w;
412         final double bottom = y + radius;
413 
414         path.reset();
415 
416         // Upper edge.
417         path.moveTo(x, bottom);
418         path.quadTo(x, y, x + radius, y);
419         path.lineTo(right - radius, y);
420         path.quadTo(right, y, right, bottom);
421 
422         // Lower edge.
423         path.lineTo(right - 1, bottom);
424         path.quadTo(right - 2, y + 2, right - radius - 1, y + 2);
425         path.lineTo(x + radius + 1, y + 2);
426         path.quadTo(x + 2, y + 2, x + 1, bottom);
427 
428         path.closePath();
429 
430         return path;
431     }

433     /**
434      * Return a path for a focus rectangle.
435      *
436      * <p>This path is suitable for filling.</p>
437      *
438      * @param  x the X coordinate of the upper-left corner of the rectangle
439      * @param  y the Y coordinate of the upper-left corner of the rectangle
440      * @param  w the width of the rectangle
441      * @param  h the height of the rectangle
442      *
443      * @return a path representing the shape.
444      */
445     public Shape createFillableFocusRectangle(int x, int y, int w, int h) {
446         final int left   = x;
447         final int top    = y;
448         final int right  = x + w;
449         final int bottom = y + h;
450 
451         path.reset();
452         path.moveTo(left, top);
453         path.lineTo(left, bottom);
454         path.lineTo(right, bottom);
455         path.lineTo(right, top);
456 
457         final float offset = 1.4f;
458 
459         final float left2   = left + offset;
460         final float top2    = top + offset;
461         final float right2  = right - offset;
462         final float bottom2 = bottom - offset;
463 
464         // TODO These two lines were curveTo in Nimbus. Perhaps we should
465         // revisit this?
466         path.lineTo(right2, top);
467         path.lineTo(right2, bottom2);
468         path.lineTo(left2, bottom2);
469         path.lineTo(left2, top2);
470         path.lineTo(right2, top2);
471         path.lineTo(right2, top);
472         path.closePath();
473 
474         return path;
475     }

516     /**
517      * Return a path for a discrete slider thumb's concentric sections.
518      *
519      * @param  x    the X coordinate of the upper-left corner of the section
520      * @param  y    the Y coordinate of the upper-left corner of the section
521      * @param  w    the width of the section
522      * @param  h    the height of the section
523      * @param  size the CornerSize representing the rounding amount for the
524      *              section
525      *
526      * @return a path representing the shape.
527      */
528     public Shape createSliderThumbDiscrete(final int x, final int y, final int w, final int h, final CornerSize size) {
529         final double topArc     = size.getRadius(w, h);
530         final double bottomArcH = size == CornerSize.INTERIOR ? 0 : 1;
531         final double bottomArcW = 3;
532 
533         path.reset();
534         path.moveTo(x, y + topArc);
535         path.quadTo(x, y, x + topArc, y);
536         path.lineTo(x + w - topArc, y);
537         path.quadTo(x + w, y, x + w, y + topArc);
538         path.lineTo(x + w, y + h / 2.0);
539         path.quadTo(x + w - bottomArcW, y + h - bottomArcH, x + w / 2.0, y + h);
540         path.quadTo(x + bottomArcW, y + h - bottomArcH, x, y + h / 2.0);
541         path.closePath();
542 
543         return path;
544     }

546     /**
547      * Return a path for a "cancel" icon. This is a circle with a punched out
548      * "x" in it.
549      *
550      * @param  x the X coordinate of the upper-left corner of the icon
551      * @param  y the Y coordinate of the upper-left corner of the icon
552      * @param  w the width of the icon
553      * @param  h the height of the icon
554      *
555      * @return a path representing the shape.
556      */
557     public Shape createCancelIcon(int x, int y, int w, int h) {
558         final double xMid = x + w / 2.0;
559         final double yMid = y + h / 2.0;
560 
561         // Draw the circle.
562         path.reset();
563         path.moveTo(xMid, y);
564         path.quadTo(x, y, x, yMid);
565         path.quadTo(x, y + h, xMid, y + h);
566         path.quadTo(x + w, y + h, x + w, yMid);
567         path.quadTo(x + w, y, xMid, y);
568         path.closePath();
569 
570         final double xOffsetL = w / 2.0 - 3;
571         final double xOffsetS = w / 2.0 - 4;
572         final double yOffsetL = h / 2.0 - 3;
573         final double yOffsetS = h / 2.0 - 4;
574         final double offsetC  = 1.5;
575 
576         // Erase the "x" with an inner subpath.
577         path.moveTo(xMid, yMid - offsetC);
578         path.lineTo(xMid + xOffsetS, yMid - yOffsetL);
579         path.lineTo(yMid + xOffsetL, yMid - yOffsetS);
580         path.lineTo(xMid + offsetC, yMid);
581         path.lineTo(xMid + xOffsetL, yMid + yOffsetS);
582         path.lineTo(xMid + xOffsetS, yMid + yOffsetL);
583         path.lineTo(xMid, yMid + offsetC);
584         path.lineTo(xMid - xOffsetS, yMid + yOffsetL);
585         path.lineTo(xMid - xOffsetL, yMid + yOffsetS);
586         path.lineTo(xMid - offsetC, yMid);
587         path.lineTo(xMid - xOffsetL, yMid - yOffsetS);
588         path.lineTo(xMid - xOffsetS, yMid - yOffsetL);
589         path.closePath();
590 
591         return path;
592     }

764      * @param  radius      the radius (arc size) used for rounding
765      * @param  topLeft     the CornerStyle of the upper-left corner. This must
766      *                     be one of {@code CornerStyle.SQUARE} or

772      *                     be one of {@code CornerStyle.SQUARE} or
773      *                     {@code CornerStyle.ROUNDED}.
774      * @param  topRight    the CornerStyle of the upper-right corner. This must
775      *                     be one of {@code CornerStyle.SQUARE} or
776      *                     {@code CornerStyle.ROUNDED}.
777      *
778      * @return a path representing the shape.
779      */
780     private Shape createRoundRectangleInternal(final int x, final int y, final int w, final int h, final double radius,
781             final CornerStyle topLeft, final CornerStyle bottomLeft, final CornerStyle bottomRight, final CornerStyle topRight) {
782         // Convenience variables.
783         final int left   = x;
784         final int top    = y;
785         final int right  = x + w;
786         final int bottom = y + h;
787 
788         // Start the path.
789         path.reset();
790 
791         // Move to top left and draw rounded corner if requested.
792         switch (topLeft) {
793 
794         case SQUARE:
795             path.moveTo(left, top);
796             break;
797 
798         case ROUNDED:
799             path.moveTo(left + radius, top);
800             path.quadTo(left, top, left, top + radius);
801             break;
802         }
803 
804         // Draw through bottom left corner.
805         switch (bottomLeft) {
806 
807         case SQUARE:
808             path.lineTo(left, bottom);
809             break;
810 
811         case ROUNDED:
812             path.lineTo(left, bottom - radius);
813             path.quadTo(left, bottom, left + radius, bottom);
814             break;
815         }
816 
817         // Draw through bottom right corner.
818         switch (bottomRight) {
819 
820         case SQUARE:
821             path.lineTo(right, bottom);
822             break;
823 
824         case ROUNDED:
825             path.lineTo(right - radius, bottom);
826             path.quadTo(right, bottom, right, bottom - radius);
827         }
828 
829         // Draw through top right corner.
830         switch (topRight) {
831 
832         case SQUARE:
833             path.lineTo(right, top);
834             break;
835 
836         case ROUNDED:
837             path.lineTo(right, top + radius);
838             path.quadTo(right, top, right - radius, top);
839             break;
840         }
841 
842         // Close the path.
843         path.closePath();
844 
845         return path;
