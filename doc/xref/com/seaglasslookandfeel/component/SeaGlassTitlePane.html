251      *
252      * @param iconify {@code true} if we are to iconify the window,
253      *                {@code false} if we are to restore the window.

255     private void setParentIcon(boolean iconify) {
256         if (rootParent instanceof JFrame) {
257             JFrame frame = (JFrame) rootParent;
258             int    state = frame.getExtendedState();
259 
260             ((JFrame) rootParent).setExtendedState(iconify ? state | Frame.ICONIFIED : state & ~Frame.ICONIFIED);

523     /**
524      * Paint the title pane.
525      *
526      * @param context the SynthContext to use to get the component.
527      * @param g       the Graphics context to paint with.
528      */
529     private void paint(SeaGlassContext context, Graphics g) {
530         String title = getTitle();
531 
532         if (title != null) {
533             SynthStyle style = context.getStyle();
534 
535             Color color = style.getColor(context, ColorType.TEXT_FOREGROUND);
536 
537             // TODO style.getColor returns improper color for state? Why?
538             if ((context.getComponentState() & 512) != 0) {
539                 Object obj = style.get(context, "[WindowFocused].textForeground");
540 
541                 if (obj != null && obj instanceof Color) {
542                     color = (Color) obj;
543                 }
544 
545                 // FIXME The state *still* doesn't get the color right!!!
546                 color = Color.BLACK;
547             }
548 
549             g.setColor(color);
550             g.setFont(style.getFont(context));
551 
552             // Center text vertically.
553             FontMetrics fm         = SwingUtilities2.getFontMetrics(rootPane, g);
554             int         baseline   = (getHeight() + fm.getAscent() - fm.getLeading() - fm.getDescent()) / 2;
555             JButton     lastButton = null;
556 
557             if (isParentIconifiable()) {
558                 lastButton = iconButton;
559             } else if (isParentMaximizable()) {
560                 lastButton = maxButton;
561             } else if (isParentClosable()) {
562                 lastButton = closeButton;
563             }
564 
565             int     maxX;
566             int     minX;
567             boolean ltr = isParentLeftToRight();
568 
569             if (ltr) {
570 
571                 if (lastButton != null) {
572                     maxX = lastButton.getX() - titleSpacing;
573                 } else {
574                     maxX = getParentWidth() - getParentInsets().right - titleSpacing;
575                 }
576 
577                 minX = titleSpacing;
578             } else {
579 
580                 if (lastButton != null) {
581                     minX = lastButton.getX() + lastButton.getWidth() + titleSpacing;
582                 } else {
583                     minX = getParentInsets().left + titleSpacing;
584                 }
585 
586                 maxX = getParentWidth() - getParentInsets().right - titleSpacing;
587             }
588 
589             String clippedTitle = getTitle(title, fm, maxX - minX);
590 
591             if (clippedTitle == title) {
592                 int width = style.getGraphicsUtils(context).computeStringWidth(context, g.getFont(), fm, title);
593 
594                 minX = Math.max(minX, (getWidth() - width) / 2);
595                 minX = Math.min(maxX - width, minX);
596             }
597 
598             // style.getGraphicsUtils(context).paintText(context, g,
599             // clippedTitle, minX, baseline - fm.getAscent(), -1);
600             ((SeaGlassGraphicsUtils) style.getGraphicsUtils(context)).drawEmphasizedText(g, color, DEFAULT_EMPHASIS_COLOR, clippedTitle,
601                                                                                          minX, baseline); // - fm.getAscent());
602         }
603     }

681     /**
682      * @see java.beans.PropertyChangeListener#propertyChange(java.beans.PropertyChangeEvent)
683      */
684     public void propertyChange(PropertyChangeEvent evt) {
685         if (evt.getSource() == this) {
686 
687             if (SeaGlassLookAndFeel.shouldUpdateStyle(evt)) {
688                 updateStyle(this);
689             }
690         }
691 
692         // Basic (from Handler inner class)
693         String prop = (String) evt.getPropertyName();
694 
695         if (closeButton != null && WINDOW_DOCUMENT_MODIFIED.equals(prop)) {
696             closeButton.revalidate();
697             closeButton.repaint();
698 
699             return;
700         }
701 
702         if (prop == JInternalFrame.IS_SELECTED_PROPERTY) {
703             repaint();
704 
705             return;
706         }
707 
708         // Frame.state isn't currently bound.
709         if ("resizable".equals(prop) || "state".equals(prop)) {
710             Frame frame = (JFrame) rootParent;
711 
712             if (frame != null) {
713                 setState(frame.getExtendedState(), true);
714             }
715 
716             if ("resizable".equals(prop)) {
717                 getRootPane().repaint();
718             }
719         } else if ("title".equals(prop)) {
720             repaint();
721         } else if ("componentOrientation" == prop) {
722             revalidate();
723             repaint();
724         } else if ("iconImage" == prop) {
725             revalidate();
726             repaint();
727         }
728 
729         if (prop == JInternalFrame.IS_ICON_PROPERTY || prop == JInternalFrame.IS_MAXIMUM_PROPERTY) {
730             setButtonTooltips();
731             enableActions();
732 
733             return;
734         }
735 
736         if ("closable" == prop) {
737 
738             if ((Boolean) evt.getNewValue() == Boolean.TRUE) {
739                 add(closeButton);
740             } else {
741                 remove(closeButton);
742             }
743         } else if ("maximizable" == prop) {
744 
745             if ((Boolean) evt.getNewValue() == Boolean.TRUE) {
746                 add(maxButton);
747             } else {
748                 remove(maxButton);
749             }
750         } else if ("iconable" == prop) {
751 
752             if ((Boolean) evt.getNewValue() == Boolean.TRUE) {
753                 add(iconButton);
754             } else {
755                 remove(iconButton);
756             }
757         }
758

885         public Dimension minimumLayoutSize(Container c) {
886             SeaGlassContext context = getContext(SeaGlassTitlePane.this);

890             int       buttonCount = 0;

895                 width  += pref.width;

911                 buttonCount++;
912             }
913 
914             FontMetrics        fm            = getFontMetrics(getFont());
915             SynthGraphicsUtils graphicsUtils = context.getStyle().getGraphicsUtils(context);
916             String             frameTitle    = getTitle();
917             int                title_w       = frameTitle != null ? graphicsUtils.computeStringWidth(context, fm.getFont(), fm, frameTitle)
918                                                                   : 0;
919             int                title_length  = frameTitle != null ? frameTitle.length() : 0;
920 
921             // Leave room for three characters in the title.
922             if (title_length > 3) {
923                 int subtitle_w = graphicsUtils.computeStringWidth(context, fm.getFont(), fm, frameTitle.substring(0, 3) + "...");
924 
925                 width += (title_w < subtitle_w) ? title_w : subtitle_w;
926             } else {
927                 width += title_w;
928             }
929 
930             height = Math.max(fm.getHeight(), height);
931 
932             width += titleSpacing + titleSpacing;
933 
934             Insets insets = getInsets();
935 
936             height += insets.top + insets.bottom;
937             width  += insets.left + insets.right;
938             context.dispose();
939 
940             return new Dimension(width, height);

943         /**
944          * Determine the position of a button.
945          *
946          * @param  c        the button.
947          * @param  insets   the title pane insets.
948          * @param  x        the x position of the button.
949          * @param  trailing Are we at the right edge?
950          *
951          * @return the modified x position for the button.
952          */
953         private int center(Component c, Insets insets, int x, boolean trailing) {
954             Dimension pref  = c.getPreferredSize();
955             int       width = pref.width;
956 
957             if (c instanceof JButton && ((JButton) c).getIcon() != null) {
958                 width = ((JButton) c).getIcon().getIconWidth();
959             }
960 
961             if (trailing) {
962                 x -= width;
963             }
964 
965             int y = 1;
966 
967             c.setBounds(x, y, pref.width, pref.height);
968 
969             if (pref.width > 0) {
970 
971                 if (!trailing) {
972                     return x + width;
973                 }
974             }
975 
976             return x;
977         }
