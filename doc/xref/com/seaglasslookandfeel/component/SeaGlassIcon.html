40  /**
41   * An icon that delegates to a painter.
42   *
43   * <p>Based on NimbusIcon by Richard Bair. Reimplemented because too much is
44   * package local.</p>
45   *
46   * @see com.sun.java.swing.plaf.nimbus.NimbusIcon
47   */
48  public class SeaGlassIcon extends SynthIcon {
49      private int    width;
50      private int    height;
51      private String prefix;
52      private String key;
53  
54      /**
55       * Creates a new SeaGlassIcon object.
56       *
57       * @param prefix the prefix to use to get the painter.
58       * @param key    the key to use to get the painter.
59       * @param w      the icon width.
60       * @param h      the icon height.
61       */
62      public SeaGlassIcon(String prefix, String key, int w, int h) {
63          this.width  = w;
64          this.height = h;
65          this.prefix = prefix;
66          this.key    = key;
67      }

69      /**
70       * Paints the icon at the specified location.
71       *
72       * @param context Identifies hosting region, may be null.
73       * @param g       the Graphics context to paint with.
74       * @param x       x location to paint to.
75       * @param y       y location to paint to.
76       * @param w       Width of the region to paint to, may be 0.
77       * @param h       Height of the region to paint to, may be 0.
78       */
79      @SuppressWarnings("unchecked")
80      @Override
81      public void paintIcon(SynthContext context, Graphics g, int x, int y, int w, int h) {
82          Painter painter = null;
83  
84          if (context != null) {
85              painter = (Painter) context.getStyle().get(context, key);
86          }
87  
88          if (painter == null) {
89              painter = (Painter) UIManager.get(prefix + "[Enabled]." + key);
90          }
91  
92          if (painter == null) {
93              painter = (Painter) UIManager.get(prefix + "." + key);
94          }
95  
96          if (painter != null && context != null) {
97              JComponent c      = context.getComponent();
98              boolean    rotate = false;
99              boolean    flip   = false;
100 
101             // translatex and translatey are additional translations that
102             // must occur on the graphics context when rendering a toolbar
103             // icon
104             int translatex = 0;
105             int translatey = 0;
106 
107             if (c instanceof JToolBar) {
108                 JToolBar toolbar = (JToolBar) c;
109 
110                 rotate = toolbar.getOrientation() == JToolBar.VERTICAL;
111                 flip   = !toolbar.getComponentOrientation().isLeftToRight();
112                 Object o = SeaGlassLookAndFeel.resolveToolbarConstraint(toolbar);
113 
114                 // we only do the +1 hack for UIResource borders, assuming
115                 // that the border is probably going to be our border
116                 if (toolbar.getBorder() instanceof UIResource) {
117 
118                     if (o == BorderLayout.SOUTH) {
119                         translatey = 1;
120                     } else if (o == BorderLayout.EAST) {
121                         translatex = 1;
122                     }
123                 }
124             }
125 
126             if (g instanceof Graphics2D) {
127                 Graphics2D gfx = (Graphics2D) g;
128 
129                 gfx.translate(x, y);
130                 gfx.translate(translatex, translatey);
131 
132                 if (rotate) {
133                     gfx.rotate(Math.toRadians(90));
134                     gfx.translate(0, -w);
135                     painter.paint(gfx, context.getComponent(), h, w);
136                     gfx.translate(0, w);
137                     gfx.rotate(Math.toRadians(-90));
138                 } else if (flip) {
139                     gfx.scale(-1, 1);
140                     gfx.translate(-w, 0);
141                     painter.paint(gfx, context.getComponent(), w, h);
142                     gfx.translate(w, 0);
143                     gfx.scale(-1, 1);
144                 } else {
145                     painter.paint(gfx, context.getComponent(), w, h);
146                 }
147 
148                 gfx.translate(-translatex, -translatey);
149                 gfx.translate(-x, -y);
150             } else {
151 
152                 // use image if we are printing to a Java 1.1 PrintGraphics as
153                 // it is not a instance of Graphics2D
154                 BufferedImage img = new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB);
155                 Graphics2D    gfx = img.createGraphics();
156 
157                 if (rotate) {
158                     gfx.rotate(Math.toRadians(90));
159                     gfx.translate(0, -w);
160                     painter.paint(gfx, context.getComponent(), h, w);
161                 } else if (flip) {
162                     gfx.scale(-1, 1);
163                     gfx.translate(-w, 0);
164                     painter.paint(gfx, context.getComponent(), w, h);
165                 } else {
166                     painter.paint(gfx, context.getComponent(), w, h);
167                 }
168 
169                 gfx.dispose();
170                 g.drawImage(img, x, y, null);
171                 img = null;
172             }
173         }
174     }

176     /**
177      * Implements the standard Icon interface's paintIcon method as the standard
178      * synth stub passes null for the context and this will cause us to not
179      * paint any thing, so we override here so that we can paint the enabled
180      * state if no synth context is available
181      *
182      * @param c the component to paint.
183      * @param g the Graphics context to paint with.
184      * @param x the x coordinate of the upper-left corner of the icon.
185      * @param y the y coordinate of the upper-left corner of the icon.
186      */
187     @SuppressWarnings("unchecked")
188     @Override
189     public void paintIcon(Component c, Graphics g, int x, int y) {
190         Painter painter = (Painter) UIManager.get(prefix + "[Enabled]." + key);
191 
192         if (painter != null) {
193             JComponent jc  = (c instanceof JComponent) ? (JComponent) c : null;
194             Graphics2D gfx = (Graphics2D) g;
195 
196             gfx.translate(x, y);
197             painter.paint(gfx, jc, width, height);
198             gfx.translate(-x, -y);
199         }
200     }

202     /**
203      * Returns the icon's width. This is a cover methods for <code>
204      * getIconWidth(null)</code>.
205      *
206      * @param  context the SynthContext describing the component/region, the
207      *                 style, and the state.
208      *
209      * @return an int specifying the fixed width of the icon.
210      */
211     @Override
212     public int getIconWidth(SynthContext context) {
213         if (context == null) {
214             return width;
215         }
216 
217         JComponent c = context.getComponent();
218 
219         if (c instanceof JToolBar && ((JToolBar) c).getOrientation() == JToolBar.VERTICAL) {
220 
221             // we only do the -1 hack for UIResource borders, assuming
222             // that the border is probably going to be our border
223             if (c.getBorder() instanceof UIResource) {
224                 return c.getWidth() - 1;
225             } else {
226                 return c.getWidth();
227             }
228         } else {
229             return scale(context, width);
230         }
231     }
232 
233     /**
234      * Returns the icon's height. This is a cover method for <code>
235      * getIconHeight(null)</code>.
236      *
237      * @param  context the SynthContext describing the component/region, the
238      *                 style, and the state.
239      *
240      * @return an int specifying the fixed height of the icon.
241      */
242     @Override
243     public int getIconHeight(SynthContext context) {
244         if (context == null) {
245             return height;
246         }
247 
248         JComponent c = context.getComponent();
249 
250         if (c instanceof JToolBar) {
251             JToolBar toolbar = (JToolBar) c;
252 
253             if (toolbar.getOrientation() == JToolBar.HORIZONTAL) {
254 
255                 // we only do the -1 hack for UIResource borders, assuming
256                 // that the border is probably going to be our border
257                 if (toolbar.getBorder() instanceof UIResource) {
258                     return c.getHeight() - 1;
259                 } else {
260                     return c.getHeight();
261                 }
262             } else {
263                 return scale(context, width);
264             }
265         } else {
266             return scale(context, height);
267         }
268     }

270     /**
271      * Scale a size based on the "JComponent.sizeVariant" client property of the
272      * component that is using this icon
273      *
274      * @param  context The synthContext to get the component from
275      * @param  size    The size to scale
276      *
277      * @return The scaled size or original if "JComponent.sizeVariant" is not
278      *         set
279      */
280     private int scale(SynthContext context, int size) {
281         if (context == null || context.getComponent() == null) {
282             return size;
283         }
284 
285         // The key "JComponent.sizeVariant" is used to match Apple's LAF
286         String scaleKey = (String) context.getComponent().getClientProperty("JComponent.sizeVariant");
287 
288         if (scaleKey != null) {
289 
290             if (SeaGlassStyle.LARGE_KEY.equals(scaleKey)) {
291                 size *= SeaGlassStyle.LARGE_SCALE;
292             } else if (SeaGlassStyle.SMALL_KEY.equals(scaleKey)) {
293                 size *= SeaGlassStyle.SMALL_SCALE;
294             } else if (SeaGlassStyle.MINI_KEY.equals(scaleKey)) {
295 
296                 // mini is not quite as small for icons as full mini is
297                 // just too tiny
298                 size *= SeaGlassStyle.MINI_SCALE + 0.07;
299             }
300         }
301 
302         return size;
303     }
