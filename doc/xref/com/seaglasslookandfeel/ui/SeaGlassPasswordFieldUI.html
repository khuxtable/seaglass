39  /**
40   * Sea Glass PasswordField UI delegate.
41   *
42   * <p>Based on SynthPasswordTextFieldUI, but needs to extend SeaGlassTextFieldUI
43   * instead.</p>
44   */
45  public class SeaGlassPasswordFieldUI extends SeaGlassTextFieldUI {
46  
47      /**
48       * {@inheritDoc}
49       */
50      @Override
51      protected void installDefaults() {
52          super.installDefaults();
53          Character echoChar = (Character) UIManager.get("PasswordField.echoChar");
54          if (echoChar != null) {
55              LookAndFeel.installProperty(getComponent(), "echoChar", echoChar);    
56          }
57      }
58  
59      /**
60       * Creates a UI for a JPasswordField.
61       *
62       * @param  c the JPasswordField
63       *
64       * @return the UI
65       */
66      public static ComponentUI createUI(JComponent c) {
67          return new SeaGlassPasswordFieldUI();
68      }
69  
70      /**
71       * Fetches the name used as a key to look up properties through the
72       * UIManager. This is used as a prefix to all the standard text properties.
73       *
74       * @return the name ("PasswordField")
75       */
76      protected String getPropertyPrefix() {
77          return "PasswordField";
78      }
79  
80      /**
81       * Creates a view (PasswordView) for an element.
82       *
83       * @param  elem the element
84       *
85       * @return the view
86       */
87      public View create(Element elem) {
88          return new PasswordView(elem);
89      }
90  
91      /**
92       * @see com.seaglasslookandfeel.ui.SeaGlassTextFieldUI#paintBackground(com.seaglasslookandfeel.SeaGlassContext,
93       *      java.awt.Graphics, javax.swing.JComponent)
94       */
95      void paintBackground(SeaGlassContext context, Graphics g, JComponent c) {
96          context.getPainter().paintPasswordFieldBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
97      }
98  
99      /**
100      * @see com.seaglasslookandfeel.ui.SeaGlassTextFieldUI#paintBorder(javax.swing.plaf.synth.SynthContext,
101      *      java.awt.Graphics, int, int, int, int)
102      */
103     public void paintBorder(SynthContext context, Graphics g, int x, int y, int w, int h) {
104         ((SeaGlassContext) context).getPainter().paintPasswordFieldBorder(context, g, x, y, w, h);
105     }
106 
107     /**
108      * @see javax.swing.plaf.basic.BasicTextUI#installKeyboardActions()
109      */
110     protected void installKeyboardActions() {
111         super.installKeyboardActions();
112         ActionMap map = SwingUtilities.getUIActionMap(getComponent());
113 
114         if (map != null && map.get(DefaultEditorKit.selectWordAction) != null) {
115             Action a = map.get(DefaultEditorKit.selectLineAction);
116 
117             if (a != null) {
118                 map.put(DefaultEditorKit.selectWordAction, a);
119             }
120         }
121     }
