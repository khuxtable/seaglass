223     /**
224      * Update the Synth styles when something changes.
225      *
226      * @param c the component.
227      */
228     private void updateStyle(JTabbedPane c) {
229         SeaGlassContext context  = getContext(c, ENABLED);
230         SynthStyle      oldStyle = style;
231 
232         style = SeaGlassLookAndFeel.updateStyle(context, this);
233 
234         tabPlacement = tabPane.getTabPlacement();
235         orientation  = ControlOrientation.getOrientation(tabPlacement == LEFT || tabPlacement == RIGHT ? VERTICAL : HORIZONTAL);
236 
237         closeButtonArmedIndex = -1;
238         Object o = c.getClientProperty("JTabbedPane.closeButton");
239 
240         if (o != null && "left".equals(o)) {
241             tabCloseButtonPlacement = LEFT;
242         } else if (o != null && "right".equals(o)) {
243             tabCloseButtonPlacement = RIGHT;
244         } else {
245             tabCloseButtonPlacement = CENTER;
246         }
247 
248         closeButtonSize   = style.getInt(context, "closeButtonSize", 6);
249         closeButtonInsets = (Insets) style.get(context, "closeButtonInsets");
250         if (closeButtonInsets == null) {
251             closeButtonInsets = new Insets(2, 2, 2, 2);
252         }
253 
254         o = c.getClientProperty("JTabbedPane.closeListener");
255         if (o != null && o instanceof SeaGlassTabCloseListener) {
256             if (tabCloseListener == null) {
257                 tabCloseListener = (SeaGlassTabCloseListener) o;
258             }
259         }
260 
261         // Add properties other than JComponent colors, Borders and
262         // opacity settings here:
263         if (style != oldStyle) {
264 
265             tabRunOverlay        = 0;
266             textIconGap          = style.getInt(context, "TabbedPane.textIconGap", 0);
267             selectedTabPadInsets = (Insets) style.get(context, "TabbedPane.selectedTabPadInsets");
268 
269             if (selectedTabPadInsets == null) {
270                 selectedTabPadInsets = new Insets(0, 0, 0, 0);
271             }
272 
273             if (oldStyle != null) {
274                 uninstallKeyboardActions();
275                 installKeyboardActions();
276             }
277         }
278 
279         context.dispose();
280 
281         if (tabContext != null) {
282             tabContext.dispose();
283         }
284 
285         tabContext    = getContext(c, Region.TABBED_PANE_TAB, ENABLED);
286         this.tabStyle = SeaGlassLookAndFeel.updateStyle(tabContext, this);
287         tabInsets     = tabStyle.getInsets(tabContext, null);
288 
289         if (tabCloseContext != null) {
290             tabCloseContext.dispose();
291         }
292 
293         tabCloseContext    = getContext(c, SeaGlassRegion.TABBED_PANE_TAB_CLOSE_BUTTON, ENABLED);
294         this.tabCloseStyle = SeaGlassLookAndFeel.updateStyle(tabCloseContext, this);
295 
296         if (tabAreaContext != null) {
297             tabAreaContext.dispose();
298         }
299 
300         tabAreaContext    = getContext(c, Region.TABBED_PANE_TAB_AREA, ENABLED);
301         this.tabAreaStyle = SeaGlassLookAndFeel.updateStyle(tabAreaContext, this);
302         tabAreaInsets     = tabAreaStyle.getInsets(tabAreaContext, null);
303 
304         if (tabContentContext != null) {
305             tabContentContext.dispose();
306         }
307 
308         tabContentContext    = getContext(c, Region.TABBED_PANE_CONTENT, ENABLED);
309         this.tabContentStyle = SeaGlassLookAndFeel.updateStyle(tabContentContext, this);
310         contentBorderInsets  = tabContentStyle.getInsets(tabContentContext, null);
311     }

334     /**
335      * Scroll the tab buttons backwards.
336      */
337     protected void scrollBackward() {
338         int selectedIndex = tabPane.getSelectedIndex();
339 
340         if (--selectedIndex < 0) {
341             tabPane.setSelectedIndex(tabPane.getTabCount() == 0 ? -1 : 0);
342         } else {
343             tabPane.setSelectedIndex(selectedIndex);
344         }
345 
346         tabPane.repaint();
347     }
348 
349     /**
350      * Scroll the tab buttons forwards.
351      */
352     protected void scrollForward() {
353         int selectedIndex = tabPane.getSelectedIndex();
354 
355         if (++selectedIndex >= tabPane.getTabCount()) {
356             tabPane.setSelectedIndex(tabPane.getTabCount() - 1);
357         } else {
358             tabPane.setSelectedIndex(selectedIndex);
359         }
360 
361         tabPane.repaint();
362     }

472     /**
473      * Get the state for the specified tab's close button.
474      *
475      * @param  c               the tabbed pane.
476      * @param  tabIndex        the index of the tab.
477      * @param  tabIsMousedOver TODO
478      *
479      * @return the close button state.
480      */
481     public int getCloseButtonState(JComponent c, int tabIndex, boolean tabIsMousedOver) {
482         if (!c.isEnabled()) {
483             return DISABLED;
484         } else if (tabIndex == closeButtonArmedIndex) {
485             return PRESSED;
486         } else if (tabIndex == closeButtonHoverIndex) {
487             return FOCUSED;
488         } else if (tabIsMousedOver) {
489             return MOUSE_OVER;
490         }
491 
492         return ENABLED;
493     }

573     protected int getBaseline(int tab) {
574         if (tabPane.getTabComponentAt(tab) != null || getTextViewForTab(tab) != null) {
575             return super.getBaseline(tab);
576         }
577 
578         String      title   = tabPane.getTitleAt(tab);
579         Font        font    = tabContext.getStyle().getFont(tabContext);

581         Icon        icon    = getIconForTab(tab);
582 
583         textRect.setBounds(0, 0, 0, 0);

588                                                                       SwingUtilities.TRAILING, calcRect, iconRect, textRect, textIconGap);
589 
590         return textRect.y + metrics.getAscent() + getBaselineOffset();
591     }

635     /**
636      * Paint the tab area, including the tabs.
637      *
638      * @param ss            the SynthContext.
639      * @param g             the Graphics context.
640      * @param tabPlacement  the side the tabs are on.
641      * @param selectedIndex the current selected tab index.
642      * @param tabAreaBounds the bounds of the tab area.
643      */
644     protected void paintTabArea(SeaGlassContext ss, Graphics g, int tabPlacement, int selectedIndex, Rectangle tabAreaBounds) {
645         Rectangle clipRect = g.getClipBounds();
646 
647         ss.setComponentState(SynthConstants.ENABLED);
648 
649         // Paint the tab area.
650         SeaGlassLookAndFeel.updateSubregion(ss, g, tabAreaBounds);
651         ss.getPainter().paintTabbedPaneTabAreaBackground(ss, g, tabAreaBounds.x, tabAreaBounds.y, tabAreaBounds.width,
652                                                          tabAreaBounds.height, tabPlacement);
653         ss.getPainter().paintTabbedPaneTabAreaBorder(ss, g, tabAreaBounds.x, tabAreaBounds.y, tabAreaBounds.width, tabAreaBounds.height,
654                                                      tabPlacement);
655 
656         iconRect.setBounds(0, 0, 0, 0);
657         textRect.setBounds(0, 0, 0, 0);
658 
659         if (runCount == 0) {
660             return;
661         }
662 
663         if (scrollBackwardButton.isVisible()) {
664             paintScrollButtonBackground(ss, g, scrollBackwardButton);
665         }
666 
667         if (scrollForwardButton.isVisible()) {
668             paintScrollButtonBackground(ss, g, scrollForwardButton);
669         }
670 
671         for (int i = leadingTabIndex; i <= trailingTabIndex; i++) {
672             if (rects[i].intersects(clipRect) && selectedIndex != i) {
673                 paintTab(tabContext, g, rects, i, iconRect, textRect);
674             }
675         }
676 
677         if (selectedIndex >= 0) {
678             if (rects[selectedIndex].intersects(clipRect)) {
679                 paintTab(tabContext, g, rects, selectedIndex, iconRect, textRect);
680             }
681         }
682     }

684     /**
685      * @see javax.swing.plaf.basic.BasicTabbedPaneUI#setRolloverTab(int)
686      */
687     protected void setRolloverTab(int index) {
688         int oldRolloverTab = getRolloverTab();
689 
690         super.setRolloverTab(index);
691 
692         Rectangle r = null;
693 
694         if ((oldRolloverTab >= 0) && (oldRolloverTab < tabPane.getTabCount())) {
695             r = getTabBounds(tabPane, oldRolloverTab);
696 
697             if (r != null) {
698                 tabPane.repaint(r);
699             }
700         }
701 
702         if (index >= 0) {
703             r = getTabBounds(tabPane, index);
704 
705             if (r != null) {
706                 tabPane.repaint(r);
707             }
708         }
709     }

1071      * Update the SynthContext for the tab area for a specified tab.
1072      *
1073      * @param index       the tab to update for.
1074      * @param selected    is the tab selected?
1075      * @param isMouseDown is the mouse down?
1076      * @param isMouseOver is the mouse over the tab?
1077      * @param hasFocus    do we have focus?
1078      */
1079     private void updateTabContext(int index, boolean selected, boolean isMouseDown, boolean isMouseOver, boolean hasFocus) {
1080         int state = 0;
1081 
1082         if (!tabPane.isEnabled() || !tabPane.isEnabledAt(index)) {
1083             state |= SynthConstants.DISABLED;
1084 
1085             if (selected) {
1086                 state |= SynthConstants.SELECTED;
1087             }
1088         } else if (selected) {
1089             state |= (SynthConstants.ENABLED | SynthConstants.SELECTED);
1090 
1091             if (isMouseOver && UIManager.getBoolean("TabbedPane.isTabRollover")) {
1092                 state |= SynthConstants.MOUSE_OVER;
1093             }
1094         } else if (isMouseOver) {
1095             state |= (SynthConstants.ENABLED | SynthConstants.MOUSE_OVER);
1096         } else {
1097             state =  SeaGlassLookAndFeel.getComponentState(tabPane);
1098             state &= ~SynthConstants.FOCUSED; // Don't use tabbedpane focus state.
1099         }
1100 
1101         if (hasFocus && tabPane.hasFocus()) {
1102             state |= SynthConstants.FOCUSED; // individual tab has focus
1103         }
1104 
1105         if (isMouseDown) {
1106             state |= SynthConstants.PRESSED;
1107         }
1108 
1109         tabContext.setComponentState(state);
1110     }
