65      protected void installDefaults() {
66          prevRenderer = header.getDefaultRenderer();
67          if (prevRenderer instanceof UIResource) {
68              header.setDefaultRenderer(new HeaderRenderer());
69          }
70          updateStyle(header);
71      }

73      private void updateStyle(JTableHeader c) {
74          SeaGlassContext context = getContext(c, ENABLED);
75          SynthStyle oldStyle = style;
76          style = SeaGlassLookAndFeel.updateStyle(context, this);
77          if (style != oldStyle) {
78              if (oldStyle != null) {
79                  uninstallKeyboardActions();
80                  installKeyboardActions();
81              }
82          }
83          context.dispose();
84      }
85  
86      protected void installListeners() {
87          super.installListeners();
88          header.addPropertyChangeListener(this);
89      }
90  
91      protected void uninstallDefaults() {
92          if (header.getDefaultRenderer() instanceof HeaderRenderer) {
93              header.setDefaultRenderer(prevRenderer);
94          }
95  
96          SeaGlassContext context = getContext(header, ENABLED);
97  
98          style.uninstallDefaults(context);
99          context.dispose();
100         style = null;
101     }
102 
103     protected void uninstallListeners() {
104         header.removePropertyChangeListener(this);

106     }

159     private class HeaderRenderer extends DefaultTableCellHeaderRenderer {
160         HeaderRenderer() {
161             setHorizontalAlignment(JLabel.LEADING);
162             setName("TableHeader.renderer");
163         }
164 
165         @Override
166         public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row,
167             int column) {
168 
169             boolean hasRollover = (column == getRolloverColumn());
170             if (isSelected || hasRollover || hasFocus) {
171                 SeaGlassLookAndFeel.setSelectedUI(
172                     (SeaGlassLabelUI) SeaGlassLookAndFeel.getUIOfType(getUI(), SeaGlassLabelUI.class), isSelected, hasFocus, table
173                         .isEnabled(), hasRollover);
174             } else {
175                 SeaGlassLookAndFeel.resetSelectedUI();
176             }
177 
178             // Stuff a variable into the client property of this renderer
179             // indicating the sort order, so that different rendering can be
180             // done for the header based on sorted state.
181             RowSorter rs = table == null ? null : table.getRowSorter();
182             java.util.List<? extends RowSorter.SortKey> sortKeys = rs == null ? null : rs.getSortKeys();
183             if (sortKeys != null && sortKeys.size() > 0 && sortKeys.get(0).getColumn() == table.convertColumnIndexToModel(column)) {
184                 switch (sortKeys.get(0).getSortOrder()) {
185                 case ASCENDING:
186                     putClientProperty("Table.sortOrder", "ASCENDING");
187                     break;
188                 case DESCENDING:
189                     putClientProperty("Table.sortOrder", "DESCENDING");
190                     break;
191                 case UNSORTED:
192                     putClientProperty("Table.sortOrder", "UNSORTED");
193                     break;
194                 default:
195                     throw new AssertionError("Cannot happen");
196                 }
197             } else {
198                 putClientProperty("Table.sortOrder", "UNSORTED");
199             }
200 
201             super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
202 
203             return this;
204         }
205

209                 super.setBorder(border);
210             }
211         }
