73      public void installDefaults() {
74          if (popupMenu.getLayout() == null || popupMenu.getLayout() instanceof UIResource) {
75              popupMenu.setLayout(new DefaultMenuLayout(popupMenu, BoxLayout.Y_AXIS));
76              popupMenu.setOpaque(false);
77          }
78          updateStyle(popupMenu);
79      }
80  
81      private void updateStyle(JComponent c) {
82          SeaGlassContext context = getContext(c, ENABLED);
83          Window window = SwingUtilities.getWindowAncestor(popupMenu);
84          if (PlatformUtils.isMac() && window != null) {
85              WindowUtils.makeWindowNonOpaque(window);
86          }
87          SeaGlassStyle oldStyle = style;
88          style = (SeaGlassStyle) SeaGlassLookAndFeel.updateStyle(context, this);
89          if (style != oldStyle) {
90              if (oldStyle != null) {
91                  uninstallKeyboardActions();
92                  installKeyboardActions();
93              }
94          }
95          context.dispose();
96      }
97  
98      protected void installListeners() {
99          super.installListeners();
100         popupMenu.addPropertyChangeListener(this);
101     }

103     protected void uninstallDefaults() {
104         SeaGlassContext context = getContext(popupMenu, ENABLED);
105 
106         style.uninstallDefaults(context);
107         context.dispose();
108         style = null;
109 
110         if (popupMenu.getLayout() instanceof UIResource) {
111             popupMenu.setLayout(null);
112         }
113     }
114 
115     protected void uninstallListeners() {
116         super.uninstallListeners();
117         popupMenu.removePropertyChangeListener(this);
118     }
