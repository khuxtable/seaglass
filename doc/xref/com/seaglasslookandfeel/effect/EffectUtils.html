56      /**
57       * Apply Gaussian Blur to Image
58       *
59       * @param  src    The image tp
60       * @param  dst    The destination image to draw blured src image into, null
61       *                if you want a new one created
62       * @param  radius The blur kernel radius
63       *
64       * @return The blured image
65       */
66      static BufferedImage gaussianBlur(BufferedImage src, BufferedImage dst, int radius) {
67          int width  = src.getWidth();
68          int height = src.getHeight();
69  
70          if (dst == null || dst.getWidth() != width || dst.getHeight() != height || src.getType() != dst.getType()) {
71              dst = createColorModelCompatibleImage(src);
72          }
73  
74          float[] kernel = createGaussianKernel(radius);
75  
76          if (src.getType() == BufferedImage.TYPE_INT_ARGB) {
77              int[] srcPixels = new int[width * height];
78              int[] dstPixels = new int[width * height];
79  
80              getPixels(src, 0, 0, width, height, srcPixels);
81  
82              // horizontal pass
83              blur(srcPixels, dstPixels, width, height, kernel, radius);
84  
85              // vertical pass
86              // noinspection SuspiciousNameCombination
87              blur(dstPixels, srcPixels, height, width, kernel, radius);
88  
89              // the result is now stored in srcPixels due to the 2nd pass
90              setPixels(dst, 0, 0, width, height, srcPixels);
91          } else if (src.getType() == BufferedImage.TYPE_BYTE_GRAY) {
92              byte[] srcPixels = new byte[width * height];
93              byte[] dstPixels = new byte[width * height];
94  
95              getPixels(src, 0, 0, width, height, srcPixels);
96  
97              // horizontal pass
98              blur(srcPixels, dstPixels, width, height, kernel, radius);
99  
100             // vertical pass
101             // noinspection SuspiciousNameCombination
102             blur(dstPixels, srcPixels, height, width, kernel, radius);
103 
104             // the result is now stored in srcPixels due to the 2nd pass
105             setPixels(dst, 0, 0, width, height, srcPixels);
106         } else {
107             throw new IllegalArgumentException("EffectUtils.gaussianBlur() src image is not a supported type, type=[" + src.getType()
108                                                + "]");
109         }
110 
111         return dst;
112     }

114     /**
115      * <p>Blurs the source pixels into the destination pixels. The force of the
116      * blur is specified by the radius which must be greater than 0.</p>
117      *
118      * <p>The source and destination pixels arrays are expected to be in the
119      * INT_ARGB format.</p>
120      *
121      * <p>After this method is executed, dstPixels contains a transposed and
122      * filtered copy of srcPixels.</p>
123      *
124      * @param srcPixels the source pixels
125      * @param dstPixels the destination pixels
126      * @param width     the width of the source picture
127      * @param height    the height of the source picture
128      * @param kernel    the kernel of the blur effect
129      * @param radius    the radius of the blur effect
130      */
131     private static void blur(int[] srcPixels, int[] dstPixels, int width, int height, float[] kernel, int radius) {
132         float a;
133         float r;
134         float g;
135         float b;
136 
137         int ca;
138         int cr;
139         int cg;
140         int cb;
141 
142         for (int y = 0; y < height; y++) {
143             int index  = y;
144             int offset = y * width;
145 
146             for (int x = 0; x < width; x++) {
147                 a = r = g = b = 0.0f;
148 
149                 for (int i = -radius; i <= radius; i++) {
150                     int subOffset = x + i;
151 
152                     if (subOffset < 0 || subOffset >= width) {
153                         subOffset = (x + width) % width;
154                     }
155 
156                     int   pixel      = srcPixels[offset + subOffset];
157                     float blurFactor = kernel[radius + i];
158 
159                     a += blurFactor * ((pixel >> 24) & 0xFF);
160                     r += blurFactor * ((pixel >> 16) & 0xFF);
161                     g += blurFactor * ((pixel >> 8) & 0xFF);
162                     b += blurFactor * ((pixel) & 0xFF);
163                 }
164 
165                 ca = (int) (a + 0.5f);
166                 cr = (int) (r + 0.5f);
167                 cg = (int) (g + 0.5f);
168                 cb = (int) (b + 0.5f);
169 
170                 dstPixels[index] =  ((ca > 255 ? 255 : ca) << 24) | ((cr > 255 ? 255 : cr) << 16) | ((cg > 255 ? 255 : cg) << 8)
171                         | (cb > 255 ? 255 : cb);
172                 index            += height;
173             }
174         }
175     }

177     /**
178      * <p>Blurs the source pixels into the destination pixels. The force of the
179      * blur is specified by the radius which must be greater than 0.</p>
180      *
181      * <p>The source and destination pixels arrays are expected to be in the
182      * BYTE_GREY format.</p>
183      *
184      * <p>After this method is executed, dstPixels contains a transposed and
185      * filtered copy of srcPixels.</p>
186      *
187      * @param srcPixels the source pixels
188      * @param dstPixels the destination pixels
189      * @param width     the width of the source picture
190      * @param height    the height of the source picture
191      * @param kernel    the kernel of the blur effect
192      * @param radius    the radius of the blur effect
193      */
194     static void blur(byte[] srcPixels, byte[] dstPixels, int width, int height, float[] kernel, int radius) {
195         float p;
196         int   cp;
197 
198         for (int y = 0; y < height; y++) {
199             int index  = y;
200             int offset = y * width;
201 
202             for (int x = 0; x < width; x++) {
203                 p = 0.0f;
204 
205                 for (int i = -radius; i <= radius; i++) {
206                     int subOffset = x + i;
207 
208                     // if (subOffset < 0) subOffset = 0;
209                     // if (subOffset >= width) subOffset = width-1;
210                     if (subOffset < 0 || subOffset >= width) {
211                         subOffset = (x + width) % width;
212                     }
213 
214                     int   pixel      = srcPixels[offset + subOffset] & 0xFF;
215                     float blurFactor = kernel[radius + i];
216 
217                     p += blurFactor * pixel;
218                 }
219 
220                 cp               =  (int) (p + 0.5f);
221                 dstPixels[index] =  (byte) (cp > 255 ? 255 : cp);
222                 index            += height;
223             }
224         }
225     }

227     /**
228      * Create a Gaussian kernel for the transformation.
229      *
230      * @param  radius the kernel radius.
231      *
232      * @return the Gaussian kernel.
233      */
234     protected static float[] createGaussianKernel(int radius) {
235         if (radius < 1) {
236             throw new IllegalArgumentException("Radius must be >= 1");
237         }
238 
239         float[] data = new float[radius * 2 + 1];
240 
241         float sigma          = radius / 3.0f;
242         float twoSigmaSquare = 2.0f * sigma * sigma;
243         float sigmaRoot      = (float) Math.sqrt(twoSigmaSquare * Math.PI);
244         float total          = 0.0f;
245 
246         for (int i = -radius; i <= radius; i++) {
247             float distance = i * i;
248             int   index    = i + radius;
249 
250             data[index] =  (float) Math.exp(-distance / twoSigmaSquare) / sigmaRoot;
251             total       += data[index];
252         }
253 
254         for (int i = 0; i < data.length; i++) {
255             data[i] /= total;
256         }
257 
258         return data;
259     }

264     /**
265      * <p>Returns an array of pixels, stored as integers, from a <code>
266      * BufferedImage</code>. The pixels are grabbed from a rectangular area
267      * defined by a location and two dimensions. Calling this method on an image
268      * of type different from <code>BufferedImage.TYPE_INT_ARGB</code> and
269      * <code>BufferedImage.TYPE_INT_RGB</code> will unmanage the image.</p>
270      *
271      * @param  img    the source image
272      * @param  x      the x location at which to start grabbing pixels
273      * @param  y      the y location at which to start grabbing pixels
274      * @param  w      the width of the rectangle of pixels to grab
275      * @param  h      the height of the rectangle of pixels to grab
276      * @param  pixels a pre-allocated array of pixels of size w*h; can be null
277      *
278      * @return <code>pixels</code> if non-null, a new array of integers
279      *         otherwise
280      */
281     static byte[] getPixels(BufferedImage img, int x, int y, int w, int h, byte[] pixels) {
282         if (w == 0 || h == 0) {
283             return new byte[0];
284         }
285 
286         if (pixels == null) {
287             pixels = new byte[w * h];
288         } else if (pixels.length < w * h) {
289             throw new IllegalArgumentException("pixels array must have a length >= w*h");
290         }
291 
292         int imageType = img.getType();
293 
294         if (imageType == BufferedImage.TYPE_BYTE_GRAY) {
295             Raster raster = img.getRaster();
296 
297             return (byte[]) raster.getDataElements(x, y, w, h, pixels);
298         } else {
299             throw new IllegalArgumentException("Only type BYTE_GRAY is supported");
300         }
301     }

334     /**
335      * <p>Returns an array of pixels, stored as integers, from a <code>
336      * BufferedImage</code>. The pixels are grabbed from a rectangular area
337      * defined by a location and two dimensions. Calling this method on an image
338      * of type different from <code>BufferedImage.TYPE_INT_ARGB</code> and
339      * <code>BufferedImage.TYPE_INT_RGB</code> will unmanage the image.</p>
340      *
341      * @param  img    the source image
342      * @param  x      the x location at which to start grabbing pixels
343      * @param  y      the y location at which to start grabbing pixels
344      * @param  w      the width of the rectangle of pixels to grab
345      * @param  h      the height of the rectangle of pixels to grab
346      * @param  pixels a pre-allocated array of pixels of size w*h; can be null
347      *
348      * @return <code>pixels</code> if non-null, a new array of integers
349      *         otherwise
350      */
351     public static int[] getPixels(BufferedImage img, int x, int y, int w, int h, int[] pixels) {
352         if (w == 0 || h == 0) {
353             return new int[0];
354         }
355 
356         if (pixels == null) {
357             pixels = new int[w * h];
358         } else if (pixels.length < w * h) {
359             throw new IllegalArgumentException("pixels array must have a length" + " >= w*h");
360         }
361 
362         int imageType = img.getType();
363 
364         if (imageType == BufferedImage.TYPE_INT_ARGB || imageType == BufferedImage.TYPE_INT_RGB) {
365             Raster raster = img.getRaster();
366 
367             return (int[]) raster.getDataElements(x, y, w, h, pixels);
368         }
369 
370         // Unmanages the image
371         return img.getRGB(x, y, w, h, pixels, 0, w);
372     }

407     /**
408      * <p>Returns a new <code>BufferedImage</code> using the same color model as
409      * the image passed as a parameter. The returned image is only compatible
410      * with the image passed as a parameter. This does not mean the returned
411      * image is compatible with the hardware.</p>
412      *
413      * @param  image the reference image from which the color model of the new
414      *               image is obtained
415      *
416      * @return a new <code>BufferedImage</code>, compatible with the color model
417      *         of <code>image</code>
418      */
419     public static BufferedImage createColorModelCompatibleImage(BufferedImage image) {
420         ColorModel cm = image.getColorModel();
421 
422         return new BufferedImage(cm, cm.createCompatibleWritableRaster(image.getWidth(), image.getHeight()), cm.isAlphaPremultiplied(),
423                                  null);
424     }

428      * height. That is, the returned <code>BufferedImage</code> is compatible
