7   package com.seaglasslookandfeel.state;
8   
9   import java.awt.Component;
10  import java.awt.Window;
11  
12  import javax.swing.JComponent;
13  
14  /**
15   * Is root pane's window in focused state?
16   */
17  public class RootPaneWindowFocusedState extends State {
18  
19      /**
20       * Creates a new RootPaneWindowFocusedState object.
21       */
22      public RootPaneWindowFocusedState() {
23          super("WindowFocused");
24      }
25  
26      /**
27       * {@inheritDoc}
28       */
29      public boolean isInState(JComponent c) {
30          Component parent = c;
31  
32          while (parent.getParent() != null) {
33  
34              if (parent instanceof Window) {
35                  break;
36              }
37  
38              parent = parent.getParent();
39          }
40  
41          if (parent instanceof Window) {
42              return ((Window) parent).isFocused();
43          }
44  
45          // Default to true.
46          return true;
47      }
48  }
