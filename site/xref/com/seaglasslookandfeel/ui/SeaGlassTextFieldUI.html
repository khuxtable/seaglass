117     private void updateStyle(JTextComponent c) {
118         SeaGlassContext context = getContext(c, ENABLED);
119         SynthStyle oldStyle = style;
120 
121         style = SeaGlassLookAndFeel.updateStyle(context, this);
122 
123         updateSearchStyle(c, context, getPropertyPrefix());
124 
125         if (style != oldStyle) {
126             updateStyle(c, context, getPropertyPrefix());
127 
128             if (oldStyle != null) {
129                 uninstallKeyboardActions();
130                 installKeyboardActions();
131             }
132         }
133         context.dispose();
134 
135         context = getContext(c, SeaGlassRegion.SEARCH_FIELD_FIND_BUTTON, ENABLED);
136         findStyle = SeaGlassLookAndFeel.updateStyle(context, this);
137         context.dispose();
138 
139         context = getContext(c, SeaGlassRegion.SEARCH_FIELD_CANCEL_BUTTON, ENABLED);
140         cancelStyle = SeaGlassLookAndFeel.updateStyle(context, this);
141         context.dispose();
142     }

144     /**
145      * Sea Glass code to support the search JTextField.variant.
146      * 
147      * @param c
148      *            the JTextField component.
149      * @param context
150      *            the SeaGlassContext.
151      * @param prefix
152      *            the control prefix, e.g. "TextField", "FormattedTextField", or
153      *            "PasswordField".
154      */
155     private void updateSearchStyle(JTextComponent c, SeaGlassContext context, String prefix) {
156         searchIconWidth = 0;
157         Object o = style.get(context, prefix + ".searchIconWidth");
158         if (o != null && o instanceof Integer) {
159             searchIconWidth = (Integer) o;
160         }
161 
162         popupIconWidth = 0;
163         o = style.get(context, prefix + ".popupIconWidth");
164         if (o != null && o instanceof Integer) {
165             popupIconWidth = (Integer) o;
166         }
167 
168         cancelIconWidth = 0;
169         o = style.get(context, prefix + ".cancelIconWidth");
170         if (o != null && o instanceof Integer) {
171             cancelIconWidth = (Integer) o;
172         }
173 
174         searchLeftInnerMargin = 0;
175         o = style.get(context, prefix + ".searchLeftInnerMargin");
176         if (o != null && o instanceof Integer) {
177             searchLeftInnerMargin = (Integer) o;
178         }
179 
180         searchRightInnerMargin = 0;
181         o = style.get(context, prefix + ".searchRightInnerMargin");
182         if (o != null && o instanceof Integer) {
183             searchRightInnerMargin = (Integer) o;
184         }
185 
186         placeholderColor = Color.GRAY;
187         o = style.get(context, "placeholderTextColor");
188         if (o != null && o instanceof Color) {
189             placeholderColor = (Color) o;
190         }
191 
192         Border border = c.getBorder();
193         if (border == null || border instanceof UIResource && !(border instanceof TextFieldBorder)) {
194             c.setBorder(createTextFieldBorder(context));
195         }
196 
197         if (isSearchField.isInState(c)) {
198             o = c.getClientProperty("JTextField.Search.PlaceholderText");
199             if (o != null && o instanceof String) {
200                 placeholderText = (String) o;
201             } else if (placeholderText != null) {
202                 placeholderText = null;
203             }
204 
205             o = c.getClientProperty("JTextField.Search.FindAction");
206             if (o != null && o instanceof ActionListener) {
207                 if (findAction == null) {
208                     findAction = (ActionListener) o;
209                 }
210             }
211 
212             o = c.getClientProperty("JTextField.Search.FindPopup");
213             if (o != null && o instanceof JPopupMenu) {
214                 if (findPopup == null) {
215                     findPopup = (JPopupMenu) o;
216                 }
217             }
218 
219             o = c.getClientProperty("JTextField.Search.CancelAction");
220             if (o != null && o instanceof ActionListener) {
221                 if (cancelAction == null) {
222                     cancelAction = (ActionListener) o;
223                 }
224             }
225 
226             installMouseListeners();
227         } else {
228             placeholderText = null;
229 
230             if (findAction != null) {
231                 findAction = null;
232             }
233 
234             if (findPopup != null) {
235                 findPopup = null;
236             }
237 
238             if (cancelAction != null) {
239                 cancelAction = null;
240             }
241 
242             uninstallMouseListeners();
243         }
244     }

247      * Private method to update styles.
248      * 
249      * @param c
250      *            the JTextField component.
251      * @param context
252      *            the SeaGlassContext.
253      * @param prefix
254      *            the control prefix, e.g. "TextField", "FormattedTextField", or
255      *            "PasswordField".

257     private void updateStyle(JTextComponent c, SeaGlassContext context, String prefix) {
258         SeaGlassStyle style = (SeaGlassStyle) context.getStyle();
259 
260         Color color = c.getCaretColor();
261         if (color == null || color instanceof UIResource) {
262             c.setCaretColor((Color) style.get(context, prefix + ".caretForeground"));
263         }
264 
265         Color fg = c.getForeground();
266         if (fg == null || fg instanceof UIResource) {
267             fg = style.getColorForState(context, ColorType.TEXT_FOREGROUND);
268             if (fg != null) {
269                 c.setForeground(fg);
270             }
271         }
272 
273         Object ar = style.get(context, prefix + ".caretAspectRatio");
274         if (ar instanceof Number) {
275             c.putClientProperty("caretAspectRatio", ar);
276         }
277 
278         context.setComponentState(SELECTED | FOCUSED);
279 
280         Color s = c.getSelectionColor();
281         if (s == null || s instanceof UIResource) {
282             c.setSelectionColor(style.getColor(context, ColorType.TEXT_BACKGROUND));
283         }
284 
285         Color sfg = c.getSelectedTextColor();
286         if (sfg == null || sfg instanceof UIResource) {
287             c.setSelectedTextColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
288         }
289 
290         context.setComponentState(DISABLED);
291 
292         Color dfg = c.getDisabledTextColor();
293         if (dfg == null || dfg instanceof UIResource) {
294             c.setDisabledTextColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
295         }
296 
297         Insets margin = c.getMargin();
298         if (margin == null || margin instanceof UIResource) {
299             margin = (Insets) style.get(context, prefix + ".margin");
300             if (margin == null) {
301                 // Some places assume margins are non-null.
302                 margin = SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS;
303             }
304             c.setMargin(margin);
305         }
306 
307         Caret caret = c.getCaret();
308         if (caret instanceof UIResource) {
309             Object o = style.get(context, prefix + ".caretBlinkRate");
310             if (o != null && o instanceof Integer) {
311                 Integer rate = (Integer) o;
312                 caret.setBlinkRate(rate.intValue());
313             }
314         }

543     public Dimension getPreferredSize(JComponent c) {
544         // The following code has been derived from BasicTextUI.
545         Document doc = ((JTextComponent) c).getDocument();
546         Insets i = c.getInsets();
547         Dimension d = c.getSize();
548         View rootView = getRootView((JTextComponent) c);
549 
550         if (doc instanceof AbstractDocument) {
551             ((AbstractDocument) doc).readLock();
552         }
553         try {
554             if ((d.width > (i.left + i.right)) && (d.height > (i.top + i.bottom))) {
555                 rootView.setSize(d.width - i.left - i.right, d.height - i.top - i.bottom);
556             } else if (d.width == 0 && d.height == 0) {
557                 // Probably haven't been layed out yet, force some sort of
558                 // initial sizing.
559                 rootView.setSize(Integer.MAX_VALUE, Integer.MAX_VALUE);
560             }
561             d.width = (int) Math.min((long) rootView.getPreferredSpan(View.X_AXIS) + (long) i.left + (long) i.right, Integer.MAX_VALUE);
562             d.height = (int) Math.min((long) rootView.getPreferredSpan(View.Y_AXIS) + (long) i.top + (long) i.bottom, Integer.MAX_VALUE);
563         } finally {
564             if (doc instanceof AbstractDocument) {
565                 ((AbstractDocument) doc).readUnlock();
566             }
567         }
568 
569         // Fix: The preferred width is always two pixels too small on a Mac.
570         d.width += 2;
571 
572         // We'd like our heights to be odd by default.
573         if ((d.height & 1) == 0) {
574             d.height--;
575         }
576 
577         return d;
578     }

610     protected void fireAction(ActionListener action) {
611         int modifiers = 0;
612         AWTEvent currentEvent = EventQueue.getCurrentEvent();
613         if (currentEvent instanceof InputEvent) {
614             modifiers = ((InputEvent) currentEvent).getModifiers();
615         } else if (currentEvent instanceof ActionEvent) {
616             modifiers = ((ActionEvent) currentEvent).getModifiers();
617         }
618         ActionEvent e = new ActionEvent(getComponent(), ActionEvent.ACTION_PERFORMED, getComponent().getText(), EventQueue
619             .getMostRecentEventTime(), modifiers);
620 
621         action.actionPerformed(e);
622     }

631          * Reinitialize the insets parameter with this Border's current Insets.

633          * @param c

639             if (insets == null) {
640                 insets = new Insets(0, 0, 0, 0);
641             }
642 
643             super.getBorderInsets(c, insets);

648                     insets.left += popupIconWidth;
649                 }
650                 insets.right += cancelIconWidth + searchRightInnerMargin;
651             }

657     /**
658      * Track mouse clicks and moves.
659      */
660     protected class MouseButtonListener extends MouseAdapter {
661         protected transient int currentMouseX, currentMouseY;
662 
663         public MouseButtonListener() {
664             isCancelArmed = false;
665         }
666 
667         public void mouseReleased(MouseEvent e) {
668             if (isCancelArmed) {
669                 isCancelArmed = false;
670                 if (isOverCancelButton()) {
671                     doCancel();
672                 }
673                 getComponent().repaint();
674             }
675         }
676 
677         public void mouseDragged(MouseEvent e) {
678 
679             currentMouseX = e.getX();
680             currentMouseY = e.getY();
681 
682             if (isCancelArmed && !isOverCancelButton()) {
683                 isCancelArmed = false;
684                 getComponent().repaint();
685             }
686         }
687 
688         public void mousePressed(MouseEvent e) {
689             if (!SwingUtilities.isLeftMouseButton(e) || !getComponent().isEnabled()) {
690                 return;
691             }
692 
693             currentMouseX = e.getX();
694             currentMouseY = e.getY();
695 
696             if (isOverFindButton()) {
697                 doFind();
698             }
699 
700             if (isOverCancelButton()) {
701                 isCancelArmed = true;
702                 getComponent().repaint();
703             } else if (isCancelArmed) {
704                 isCancelArmed = false;
705                 getComponent().repaint();
706             }
707         }
708 
709         private boolean isOverFindButton() {
710             return getFindButtonBounds().contains(currentMouseX, currentMouseY);
711         }
712 
713         public boolean isOverCancelButton() {
714             return getCancelButtonBounds().contains(currentMouseX, currentMouseY);
715         }
