39  public final class ProgressBarPainter extends AbstractCommonColorsPainter {
40      public static enum Which {
41          BACKGROUND_ENABLED,
42          BACKGROUND_DISABLED,
43  
44          FOREGROUND_ENABLED,
45          FOREGROUND_ENABLED_FINISHED,
46          FOREGROUND_ENABLED_INDETERMINATE,
47          FOREGROUND_DISABLED,
48          FOREGROUND_DISABLED_FINISHED,
49          FOREGROUND_DISABLED_INDETERMINATE,
50      }

52      private Color                        progressBarTrackInteriorEnabled         = decodeColor("progressBarTrackInterior");
53      private Color                        buttonInteriorBaseSelected              = decodeColor("buttonInteriorBaseSelected");
54      private Color                        progressBarTrackBase                    = decodeColor("progressBarTrackBase");
55  
56      private Color                        progressBarTrackInteriorDisabled        = disable(progressBarTrackInteriorEnabled);
57      private TwoColors                    progressBarTrackEnabled                 = new TwoColors(deriveColor(progressBarTrackBase,
58                                                                                       -0.000749f, 0.005236f, 0f, 0), progressBarTrackBase);
59      private TwoColors                    progressBarTrackDisabled                = disable(progressBarTrackEnabled);
60      private Color                        progressBarEndEnabled                   = deriveColor(buttonInteriorBaseSelected, 0.006370f,
61                                                                                       0.274445f, -0.074510f, 0);
62      private Color                        progressBarEndDisabled                  = disable(progressBarEndEnabled);
63  
64      private FourColors                   progressBarEnabled                      = getCommonInteriorColors(CommonControlState.SELECTED);
65      private FourColors                   progressBarDisabled                     = disable(progressBarEnabled);
66  
67      private FourColors                   progressBarIndeterminatePatternEnabled  = getCommonInteriorColors(CommonControlState.ENABLED);
68      private FourColors                   progressBarIndeterminatePatternDisabled = disable(progressBarIndeterminatePatternEnabled);
69  
70      private Effect                       dropShadow                              = new SeaGlassDropShadowEffect();
71      private SeaGlassInternalShadowEffect internalShadow                          = new SeaGlassInternalShadowEffect();

73      private Which                        state;
74      private PaintContext                 ctx;
75      private CommonControlState            type;

77      public ProgressBarPainter(Which state) {
78          super();
79          this.state = state;
80          this.ctx = new PaintContext(CacheMode.FIXED_SIZES);
81          type = getButtonType(state);
82      }
83  
84      @Override
85      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
86          switch (state) {
87          case BACKGROUND_ENABLED:
88              paintTrack(g, width, height);
89              break;
90          case BACKGROUND_DISABLED:
91              paintTrack(g, width, height);
92              break;
93          case FOREGROUND_ENABLED:
94              paintBar(g, width, height, false);
95              break;
96          case FOREGROUND_ENABLED_FINISHED:
97              paintBar(g, width, height, true);
98              break;
99          case FOREGROUND_ENABLED_INDETERMINATE:
100             paintIndeterminateBar(g, width, height);
101             break;
102         case FOREGROUND_DISABLED:
103             paintBar(g, width, height, false);
104             break;
105         case FOREGROUND_DISABLED_FINISHED:
106             paintBar(g, width, height, true);
107             break;
108         case FOREGROUND_DISABLED_INDETERMINATE:
109             paintIndeterminateBar(g, width, height);
110             break;
111         }
112     }

114     @Override
115     protected PaintContext getPaintContext() {
116         return ctx;

119     private CommonControlState getButtonType(Which state) {
120         switch (state) {
121         case BACKGROUND_DISABLED:
122         case FOREGROUND_DISABLED:
123         case FOREGROUND_DISABLED_FINISHED:
124         case FOREGROUND_DISABLED_INDETERMINATE:
125             return CommonControlState.DISABLED;
126         case BACKGROUND_ENABLED:
127         case FOREGROUND_ENABLED:
128         case FOREGROUND_ENABLED_FINISHED:
129         case FOREGROUND_ENABLED_INDETERMINATE:
130             return CommonControlState.ENABLED;
131         }
132         return null;
133     }
134 
135     private void paintTrack(Graphics2D g, int width, int height) {
136         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
137 
138         Shape s = shapeGenerator.createRoundRectangle(2, 2, width - 5, height - 5, CornerSize.ROUND_HEIGHT_DRAW);
139         if (state != Which.BACKGROUND_DISABLED) {
140             dropShadow.fill(g, s);
141         }
142         g.setPaint(getProgressBarBorderPaint(s, type));
143         g.draw(s);
144 
145         s = shapeGenerator.createRoundRectangle(3, 3, width - 6, height - 6, CornerSize.ROUND_HEIGHT);
146         g.setPaint(getProgressBarTrackPaint(s, type));
147         g.fill(s);
148 
149         s = shapeGenerator.createInternalDropShadowRounded(3, 3, width - 6, height - 8);
150         internalShadow.fill(g, s, true, true);
151     }
152 
153     private void paintBar(Graphics2D g, int width, int height, boolean isFinished) {
154         Shape s = shapeGenerator.createRectangle(0, 0, width, height);
155         g.setPaint(getProgressBarPaint(s, type));
156         g.fill(s);
157 
158         if (!isFinished) {
159             s = shapeGenerator.createRectangle(width - 1, 0, 1, height);
160             g.setPaint(getProgressBarEndPaint(s, type));
161             g.fill(s);
162         }
163     }
164 
165     private void paintIndeterminateBar(Graphics2D g, int width, int height) {
166         Shape s = shapeGenerator.createRectangle(0, 0, width, height);
167         g.setPaint(getProgressBarPaint(s, type));
168         g.fill(s);
169         s = shapeGenerator.createProgressBarIndeterminatePattern(0, 0, width, height);
170         g.setPaint(getProgressBarIndeterminatePaint(s, type));
171         g.fill(s);
172     }
