20  package com.seaglasslookandfeel.painter;
21  
22  import java.awt.Graphics2D;
23  import java.awt.Shape;
24  
25  import javax.swing.JComponent;
26  
27  import com.seaglasslookandfeel.effect.SeaGlassInternalShadowEffect;
28  import com.seaglasslookandfeel.painter.AbstractRegionPainter.PaintContext.CacheMode;
29  
30  /**
31   * ComboBoxTextFieldPainter implementation.
32   */
33  public final class ComboBoxTextFieldPainter extends AbstractCommonColorsPainter {
34      public static enum Which {
35          BACKGROUND_DISABLED, BACKGROUND_ENABLED, BACKGROUND_SELECTED,
36      }
37  
38      private SeaGlassInternalShadowEffect internalShadow = new SeaGlassInternalShadowEffect();
39  
40      private PaintContext                 ctx;
41      private CommonControlState            type;
42  
43      public ComboBoxTextFieldPainter(Which state) {
44          super();
45          this.ctx = new PaintContext(CacheMode.FIXED_SIZES);
46          this.type = (state == Which.BACKGROUND_DISABLED) ? CommonControlState.DISABLED : CommonControlState.ENABLED;
47      }
48  
49      @Override
50      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
51          Shape s = shapeGenerator.createRectangle(3, 3, width - 3, height - 6);
52          g.setColor(c.getBackground());
53          g.fill(s);
54  
55          s = shapeGenerator.createRectangle(3, 3, width - 3, height - 6);
56          internalShadow.fill(g, s, false, false);
57  
58          s = shapeGenerator.createOpenRectangle(2, 2, width - 3, height - 5);
59          g.setPaint(getTextBorderPaint(type, false));
60          g.draw(s);
61      }
62  
63      @Override
64      protected PaintContext getPaintContext() {
65          return ctx;
66      }
67  }
