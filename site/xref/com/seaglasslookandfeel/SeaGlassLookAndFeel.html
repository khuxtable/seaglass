491     /**
492      * Initialize the default font.
493      *
494      * @return the default font.
495      */
496     private Font getDefaultFont() {
497         /*
498          * Set the default font to Lucida Grande if available, else use Lucida
499          * Sans Unicode. Grande is a later font and a bit nicer looking, but it
500          * is a derivation of Sans Unicode, so they're compatible.
501          */
502         Font font = null;
503 
504         if (PlatformUtils.isMac()) {
505             font = new Font("Lucida Grande", Font.PLAIN, 13);
506         }
507 
508         if (font == null) {
509             font = new Font("Lucida Sans Unicode", Font.PLAIN, 13);
510         }
511 
512         return font;
513     }

2061     /**
2062      * Package private method which returns either BorderLayout.NORTH,
2063      * BorderLayout.SOUTH, BorderLayout.EAST, or BorderLayout.WEST depending on
2064      * the location of the toolbar in its parent. The toolbar might be in
2065      * PAGE_START, PAGE_END, CENTER, or some other position, but will be
2066      * resolved to either NORTH,SOUTH,EAST, or WEST based on where the toolbar
2067      * actually IS, with CENTER being NORTH.
2068      *
2069      * <p/>This code is used to determine where the border line should be drawn
2070      * by the custom toolbar states, and also used by SeaGlassIcon to determine
2071      * whether the handle icon needs to be shifted to look correct.</p>
2072      *
2073      * <p>Toollbars are unfortunately odd in the way these things are handled,
2074      * and so this code exists to unify the logic related to toolbars so it can
2075      * be shared among the static files such as SeaGlassIcon and generated files
2076      * such as the ToolBar state classes.</p>
2077      *
2078      * @param  toolbar a toolbar in the Swing hierarchy.
2079      *
2080      * @return the {@code BorderLayout} orientation of the toolbar, or
2081      *         {@code BorderLayout.NORTH} if none can be determined.
2082      */
2083     public static Object resolveToolbarConstraint(JToolBar toolbar) {
2084         /*
2085          * NOTE: we don't worry about component orientation or PAGE_END etc
2086          * because the BasicToolBarUI always uses an absolute position of
2087          * NORTH/SOUTH/EAST/WEST.
2088          */
2089         if (toolbar != null) {
2090             Container parent = toolbar.getParent();
2091 
2092             if (parent != null) {
2093                 LayoutManager m = parent.getLayout();
2094 
2095                 if (m instanceof BorderLayout) {
2096                     BorderLayout b   = (BorderLayout) m;
2097                     Object       con = b.getConstraints(toolbar);
2098 
2099                     if (con == SOUTH || con == EAST || con == WEST) {
2100                         return con;
2101                     }
2102 
2103                     return NORTH;
2104                 }
2105             }
2106         }
2107 
2108         return NORTH;
2109     }
